-- | CantonCoin.MiningRoundSync - Canton Coin 10-Minute Transaction Window
-- ✅ SECTION C.3: Mining round synchronization logic
-- ✅ FIXED: Decimal/Int type conversions
-- ✅ FIXED: Controller changed to use explicit caller parameter
--           Any stakeholder (amuletApp or any observer) can call the choices
--           by passing themselves as the caller parameter.
-- ✅ FIXED: CCTransactionPrecondition.VerifyRoundAndProceed controller changed from
--           `user` alone to `user, amuletApp` — both must co-sign. This reflects the
--           real-world requirement that the amulet app co-authorises round-validity
--           checks, preventing a rogue user from forging round state.
--
-- PROBLEM: Canton Coin (CC) transactions reference an OpenMiningRound contract
-- that is only valid for 10 minutes. External signing (custody solutions like
-- Fireblocks) can take >10 minutes, causing transactions to fail.
--
-- SOLUTION: Add explicit mining round checks and user warnings before transaction
-- preparation. Give users a "time remaining" countdown.

module CantonCoin.MiningRoundSync where

import DA.Time

import TokenStandard.Interfaces

-- ✅ C.3: MINING ROUND REFERENCE
-------------------------------

-- | Represents the current Canton Coin mining round
-- In production, this would be fetched from the Amulet app
template MiningRoundReference
  with
    roundNumber   : Int
    openedAt      : Time
    expiresAt     : Time
    amuletApp     : Party
    observers     : [Party]
  where
    signatory amuletApp
    observer observers

    ensure
      expiresAt > openedAt &&
      roundNumber > 0

    -- ✅ C.3: Check if round is still valid
    -- ✅ FIXED: Controller is now `caller` with explicit parameter
    --           Any stakeholder can call this by passing themselves as caller
    nonconsuming choice IsRoundValid : Bool
      with caller : Party
      controller caller
      do
        assertMsg "Caller must be a stakeholder" (caller == amuletApp || caller `elem` observers)
        now <- getTime
        return (now < expiresAt)

    -- ✅ C.3: Get remaining time in seconds
    -- ✅ FIXED: Controller is now `caller` with explicit parameter
    -- ✅ FIXED: Return Int and use Decimal literal for division
    nonconsuming choice GetRemainingSeconds : Int
      with caller : Party
      controller caller
      do
        assertMsg "Caller must be a stakeholder" (caller == amuletApp || caller `elem` observers)
        now <- getTime
        let diff = convertRelTimeToMicroseconds (subTime expiresAt now)
        return (truncate (intToDecimal diff / 1000000.0))

    -- ✅ C.3: Check if there's enough time for external signing
    -- ✅ FIXED: Controller is now `caller` with explicit parameter
    nonconsuming choice HasTimeForExternalSigning : Bool
      with
        estimatedSigningDelaySeconds : Int
        caller : Party
      controller caller
      do
        assertMsg "Caller must be a stakeholder" (caller == amuletApp || caller `elem` observers)
        now <- getTime
        let diff = convertRelTimeToMicroseconds (subTime expiresAt now)
        let remainingSeconds = intToDecimal diff / 1000000.0
        return (remainingSeconds > intToDecimal estimatedSigningDelaySeconds)

-- ✅ C.3: TRANSACTION PREPARATION GUARD
-------------------------------

-- | Template to check before preparing CC transactions
template CCTransactionPrecondition
  with
    user              : Party
    roundReferenceCid : ContractId MiningRoundReference
    operation         : Text  -- E.g., "Transfer", "DVP Settlement", "Loan Repayment"
  where
    signatory user

    -- ✅ C.3: Verify round is valid before proceeding
    -- ✅ FIXED: Controller is now `user, amuletApp` — both must co-sign.
    --           amuletApp is looked up from the MiningRoundReference contract.
    --           This satisfies DAML's authorization requirement when the choice body
    --           exercises amuletApp-controlled choices on MiningRoundReference.
    choice VerifyRoundAndProceed : CCTransactionApproval
      with
        estimatedSigningDelaySeconds : Int  -- User's custody setup delay
        amuletApp                    : Party -- Must match MiningRoundReference.amuletApp
      controller user, amuletApp
      do
        now <- getTime

        -- Check round validity (caller = user)
        isValid <- exercise roundReferenceCid IsRoundValid with caller = user
        assertMsg "Mining round has expired. Please refresh." isValid

        -- Check sufficient time remaining (caller = user)
        hasTime <- exercise roundReferenceCid HasTimeForExternalSigning with
          estimatedSigningDelaySeconds = estimatedSigningDelaySeconds
          caller = user

        remainingSeconds <- exercise roundReferenceCid GetRemainingSeconds with caller = user

        if not hasTime then do
          -- ❌ Not enough time - return warning
          return CCTransactionApproval with
            approved = False
            warningMessage = Some ("Only " <> show remainingSeconds <> " seconds remaining in mining round. " <>
                                   "Your custody solution needs " <> show estimatedSigningDelaySeconds <> " seconds. " <>
                                   "Please wait for the next mining round to open.")
            approvedAt = now
            expiresAt = addRelTime now (seconds remainingSeconds)
        else do
          -- ✅ Enough time - proceed
          return CCTransactionApproval with
            approved = True
            warningMessage = None
            approvedAt = now
            expiresAt = addRelTime now (seconds remainingSeconds)

-- | Wrapper around standard transfer that checks CC mining round
template CCTransferWithRoundCheck
  with
    sender            : Party
    receiver          : Party
    assetId           : Text  -- Must be Canton Coin
    amount            : Decimal
    roundReferenceCid : ContractId MiningRoundReference
    senderHoldingCid  : ContractId Holding
    issuer            : Party
    amuletApp         : Party  -- Must match MiningRoundReference.amuletApp
  where
    signatory sender
    observer receiver, issuer

    choice PrepareTransferWithRoundCheck : Either Text (ContractId TransferInstruction)
      with
        estimatedSigningDelaySeconds : Int
      controller sender, amuletApp
      do
        now <- getTime

        -- ✅ C.3: Check mining round first
        approval <- create CCTransactionPrecondition with
          user = sender
          roundReferenceCid = roundReferenceCid
          operation = "CC Transfer"

        approvalResult <- exercise approval VerifyRoundAndProceed with
          estimatedSigningDelaySeconds = estimatedSigningDelaySeconds
          amuletApp = amuletApp

        if not approvalResult.approved then do
          -- Return error message
          let errMsg = fromOptional "Mining round check failed" approvalResult.warningMessage
          return (Left errMsg)
        else do
          -- ✅ Proceed with transfer
          transferCid <- create TransferInstruction_Impl with
            sender = sender
            receiver = receiver
            instrument = InstrumentId with
              depository = issuer
              issuer = issuer
              id = assetId
              version = "0"
            amount = amount
            deadline = approvalResult.expiresAt  -- ✅ C.3: Use round expiry as deadline
            status = Pending
            metadata = Map.fromList
              [ ("cantonsuite.io/transfer-type", "CC-FOP")
              , ("cantonsuite.io/round-checked", "true")
              , ("cantonsuite.io/approved-at", show now)
              ]
            senderHoldingCid = senderHoldingCid

          return (Right (toInterfaceContractId transferCid))

-- | Calculate mining round status for UI display
-- In production, this would be called from the frontend via Ledger API
-- ✅ FIXED: Renamed parameter from 'round' to 'roundRef' to avoid shadowing
calculateRoundStatus : MiningRoundReference -> Time -> MiningRoundStatus
calculateRoundStatus roundRef currentTime =
  let diff = convertRelTimeToMicroseconds (subTime roundRef.expiresAt currentTime)
      remainingSeconds = intToDecimal diff / 1000000.0
      totalDuration = 600  -- 10 minutes = 600 seconds
      percentComplete = ((intToDecimal totalDuration - remainingSeconds) / intToDecimal totalDuration) * 100.0

      -- ✅ C.3: Warn if <2 minutes remaining
      canPrepareNow = remainingSeconds > 120.0
      warningMsg = if not canPrepareNow
                   then Some ("Only " <> show remainingSeconds <> " seconds remaining. Wait for next round.")
                   else None
  in
    MiningRoundStatus with
      roundNumber = roundRef.roundNumber
      remainingSeconds = truncate remainingSeconds
      totalDuration = totalDuration
      percentComplete = percentComplete
      canPrepareNow = canPrepareNow
      warningMessage = warningMsg

-- | Compliance.AtomicAudit - 100% Atomic Regulatory Audit Enforcement
-- ✅ SECTION F.1: Atomic Regulatory Audit Enforcement
--
-- PROBLEM: RegulatorView creation happens AFTER settlement, creating a gap where
-- transactions could complete without regulatory visibility.
--
-- SOLUTION: Atomically create RegulatorView in the SAME transaction as settlement.
-- This ensures 100% audit coverage with zero gaps ("Audit Everything, See Nothing Private").

module Compliance.AtomicAudit where

import DA.Time
import DA.Text (sha256)
import DA.Map qualified as Map
import TokenStandard.Interfaces
import Portfolio (Holding_Impl(..))
import Regulator (RegulatorView(..))
import DA.Action (when)

-- ✅ F.1: ATOMIC AUDIT PATTERN
-------------------------------

-- Result type that ALWAYS includes RegulatorView
data AuditedSettlementResult = AuditedSettlementResult
  with
    buyerHolding     : ContractId Holding
    sellerHolding    : ContractId Holding
    regulatorView    : ContractId RegulatorView  -- ✅ ALWAYS PRESENT
    settlementTime   : Time
  deriving (Eq, Show)

-------------------------------
-- ✅ F.1: ATOMIC TRADE SETTLEMENT
-- Replaces ExecuteDirectSettlement in Trade.daml
-------------------------------

template AtomicTradeSettlement
  with
    buyer              : Party
    seller             : Party
    assetIssuer        : Party
    assetId            : Text
    quantity           : Decimal
    pricePerUnit       : Decimal
    complianceParty    : Party
    regulatorParty     : Party
    buyerPaymentCid    : ContractId Holding
    sellerAssetCid     : ContractId Holding
  where
    signatory buyer, seller
    observer complianceParty, assetIssuer, regulatorParty
    
    -- ✅ F.1: Settlement + Audit in ONE ATOMIC TRANSACTION
    choice ExecuteAtomicSettlement : AuditedSettlementResult
      controller assetIssuer
      do
        now <- getTime
        
        -- Verify holdings
        buyerPayment <- fetch buyerPaymentCid
        sellerAsset <- fetch sellerAssetCid
        
        let totalPrice = quantity * pricePerUnit
        assertMsg "Insufficient payment" (getAmount_H buyerPayment >= totalPrice)
        assertMsg "Insufficient assets" (getAmount_H sellerAsset >= quantity)
        
        -- Archive old holdings
        archive buyerPaymentCid
        archive sellerAssetCid
        
        -- Create new holdings for buyer (asset)
        buyerNewHolding <- create Holding_Impl with
          owner = buyer
          assetId = assetId
          quantity = quantity
          issuer = assetIssuer
          compliance = complianceParty
          locked = False
          lockContext = None
          metadata = Map.fromList 
            [ ("trade-type", "atomic-settlement")
            , ("price-per-unit", show pricePerUnit)
            , ("settlement-time", show now)
            ]
          observers   = [] 
        
        -- Create new holdings for seller (payment)
        sellerNewHolding <- create Holding_Impl with
          owner = seller
          assetId = (getInstrument_H buyerPayment).id
          quantity = totalPrice
          issuer = assetIssuer
          compliance = complianceParty
          locked = False
          lockContext = None
          metadata = Map.fromList
            [ ("trade-type", "atomic-settlement")
            , ("asset-sold", assetId)
            , ("settlement-time", show now)
            ]
          observers   = [] 

        -- Handle change for buyer if overpaid
        if getAmount_H buyerPayment > totalPrice then do
          _ <- create Holding_Impl with
            owner = buyer
            assetId = (getInstrument_H buyerPayment).id
            quantity = getAmount_H buyerPayment - totalPrice
            issuer = assetIssuer
            compliance = complianceParty
            locked = False
            lockContext = None
            metadata = Map.empty
            observers   = [] 
          pure ()
        else pure ()
        
        -- Handle leftover asset for seller
        if getAmount_H sellerAsset > quantity then do
          _ <- create Holding_Impl with
            owner = seller
            assetId = assetId
            quantity = getAmount_H sellerAsset - quantity
            issuer = assetIssuer
            compliance = complianceParty
            locked = False
            lockContext = None
            metadata = Map.empty
            observers   = [] 
          pure ()
        else pure ()
        
        -- ✅ F.1: ATOMICALLY CREATE REGULATOR VIEW (same transaction)
        let buyerHash = sha256 (show buyer)
        let sellerHash = sha256 (show seller)
        let txHash = sha256 (show buyer <> show seller <> show quantity <> show now)
        
        regulatorViewCid <- create RegulatorView with
          regulator = regulatorParty
          assetId = assetId
          quantity = quantity
          buyerHash = buyerHash
          sellerHash = sellerHash
          issuer = assetIssuer
          executedAt = now
          eventType = "TRADE"
          transactionHash = txHash
        
        -- Return result with GUARANTEED regulator view
        return AuditedSettlementResult with
          buyerHolding = toInterfaceContractId buyerNewHolding
          sellerHolding = toInterfaceContractId sellerNewHolding
          regulatorView = regulatorViewCid
          settlementTime = now

data AuditedRedemptionResult = AuditedRedemptionResult
  with
    cashHolding    : ContractId Holding
    regulatorView  : ContractId RegulatorView  -- ✅ ALWAYS PRESENT
    redemptionTime : Time
  deriving (Eq, Show)

template AtomicRedemptionApproval
  with
    redeemer       : Party
    issuer         : Party
    compliance     : Party
    regulator      : Party
    assetId        : Text
    quantity       : Decimal
    cashAmount     : Decimal
    lockedAssetCid : ContractId Holding  -- Locked via allocation
  where
    signatory redeemer, issuer
    observer compliance, regulator
    
    -- ✅ F.1: Redemption + Audit in ONE ATOMIC TRANSACTION
    choice ExecuteAtomicRedemption : AuditedRedemptionResult
      controller issuer
      do
        now <- getTime
        
        -- Verify locked holding
        lockedAsset <- fetch lockedAssetCid
        assertMsg "Asset not locked" (isLocked lockedAsset)
        assertMsg "Insufficient quantity" (getAmount_H lockedAsset >= quantity)
        
        -- Burn the locked asset
        archive lockedAssetCid
        
        -- Create cash payment holding
        cashHolding <- create Holding_Impl with
          owner = redeemer
          assetId = "USD"  -- Or specified payment currency
          quantity = cashAmount
          issuer = issuer
          compliance = compliance
          locked = False
          lockContext = None
          metadata = Map.fromList
            [ ("redemption-type", "atomic")
            , ("redeemed-asset", assetId)
            , ("redeemed-quantity", show quantity)
            , ("redemption-time", show now)
            ]
          observers   = [] 
        
        -- ✅ F.1: ATOMICALLY CREATE REGULATOR VIEW (same transaction)
        let redeemerHash = sha256 (show redeemer)
        let txHash = sha256 (show redeemer <> show quantity <> show now)
        
        regulatorViewCid <- create RegulatorView with
          regulator = regulator
          assetId = assetId
          quantity = quantity
          buyerHash = "BURN-DESTINATION"  -- No buyer in redemption
          sellerHash = redeemerHash
          issuer = issuer
          executedAt = now
          eventType = "REDEMPTION"
          transactionHash = txHash
        
        return AuditedRedemptionResult with
          cashHolding = toInterfaceContractId cashHolding
          regulatorView = regulatorViewCid
          redemptionTime = now

template AtomicBurnWithAudit
  with
    issuer     : Party
    compliance : Party
    regulator  : Party
    burnHolder : Party
    holdingCid : ContractId Holding
    quantity   : Decimal
    burnReason : Text
  where
    signatory issuer, burnHolder  -- ✅ FIX AF-3: holder added as signatory
    observer compliance, regulator
    
    choice ExecuteAtomicBurn : AuditedBurnResult
      controller issuer
      do
        now <- getTime
        
        -- 1. Fetch the actual contract
        holding <- fetch holdingCid
        let holdingAmount = getAmount_H holding -- The total balance in this contract
        let assetId = (getInstrument_H holding).id
        
        -- 2. Verify we aren't burning more than we have
        assertMsg "Burn quantity exceeds holding amount" (holdingAmount >= quantity)
        
        -- 3. Destroy the old contract
        archive holdingCid
        
        -- 4. CHANGE LOGIC: If we only burned part of the contract, 
        --    create a new holding with the leftover amount.
        when (holdingAmount > quantity) $ do
          _ <- create Holding_Impl with
            owner = burnHolder
            assetId = assetId
            quantity = holdingAmount - quantity
            issuer = issuer
            compliance = compliance
            locked = False
            lockContext = None
            metadata = Map.empty
            observers = []
          pure ()
        
        -- 5. ✅ F.1: ATOMICALLY CREATE REGULATOR VIEW
        let holderHash = sha256 (show burnHolder)
        let txHash = sha256 (show burnHolder <> show quantity <> show now)
        
        regulatorViewCid <- create RegulatorView with
          regulator = regulator
          assetId = assetId
          quantity = quantity -- Log exactly how much was burned
          buyerHash = "BURN-DESTINATION"
          sellerHash = holderHash
          issuer = issuer
          executedAt = now
          eventType = "BURN"
          transactionHash = txHash
        
        return AuditedBurnResult with
          regulatorView = regulatorViewCid
          burnTime = now

-- ✅ F.1: AUDIT COMPLETENESS CHECKER
-------------------------------

-- Verify that every settlement has a corresponding RegulatorView
template AuditCompletenessCheck
  with
    regulator      : Party
    compliance     : Party
    checkPeriodStart : Time
    checkPeriodEnd   : Time
  where
    signatory compliance
    observer regulator
    
    nonconsuming choice VerifyAuditCompleteness : AuditCompletenessReport
      with
        totalSettlements  : Int
        totalRegulatorViews : Int
        missingAudits     : Int
      controller regulator
      do
        let completeness = if totalSettlements == 0
                          then 100.0
                          else (intToDecimal totalRegulatorViews / intToDecimal totalSettlements) * 100.0
        
        let isComplete = completeness == 100.0
        
        return AuditCompletenessReport with
          periodStart = checkPeriodStart
          periodEnd = checkPeriodEnd
          totalTransactions = totalSettlements
          auditedTransactions = totalRegulatorViews
          missingAudits = missingAudits
          completenessPercentage = completeness
          fullyCompliant = isComplete

data AuditCompletenessReport = AuditCompletenessReport
  with
    periodStart           : Time
    periodEnd             : Time
    totalTransactions     : Int
    auditedTransactions   : Int
    missingAudits         : Int
    completenessPercentage: Decimal
    fullyCompliant        : Bool
  deriving (Eq, Show)

// ===== daml/Compliance/SanctionsRegistry.daml =====

-- | Compliance.SanctionsRegistry - AML/Sanctions Clearance via Commitment Pattern
-- ✅ NEW FILE: Implements Task 1.4 (Proactive AML Blacklist)
--
-- DESIGN: Commitment pattern (NOT an on-ledger list that DVP observes).
--
-- PROBLEM WITH NAIVE APPROACH: If SanctionsRegistry is an observer on DVPSettlement,
-- the full sanctions list is visible to all DVP participants — a severe privacy leak.
--
-- SOLUTION: Compliance performs the check off-ledger, then issues a per-trade
-- SanctionsClearance voucher (on-ledger proof). DVP requires clearances as inputs.
-- The full sanctions list NEVER appears on the DVP observer set. Only the compliance
-- party ever sees the registry. Clearances are short-lived and single-use.
--
-- INTEGRATION: DVP_Settlement.ExecuteAtomicDVP requires buyerClearanceCid and
-- sellerClearanceCid. Both clearances are consumed atomically with settlement.
--
-- ✅ FIXED: Removed explicit `archive self` from AddSanctionedEntity and
--           RemoveSanctionedEntity. DAML consuming choices archive the contract
--           implicitly — calling `archive self` inside a consuming choice archives
--           it a second time mid-transaction, causing "consumed in same transaction".

module Compliance.SanctionsRegistry where

import DA.Text (sha256)
import DA.Time

-------------------------------
-- SANCTIONS REGISTRY
-- The hashed registry. Never appears on DVP observer sets.
-------------------------------

template SanctionsRegistry
  with
    compliance       : Party
    regulator        : Party
    sanctionedHashes : [Text]  -- SHA-256(show party) for each sanctioned entity
    lastUpdated      : Time
    listVersion      : Text
  where
    signatory compliance
    observer regulator

    -- ✅ FIXED: Removed `archive self`. Consuming choices archive implicitly in DAML.
    --           The old code did: archive self → create this with ...
    --           That caused "Attempt to exercise a contract that was consumed in same transaction"
    --           because `archive self` fired first, then the implicit archive fired again.
    choice AddSanctionedEntity : ContractId SanctionsRegistry
      with
        partyHash : Text
        reason    : Text
      controller compliance
      do
        now <- getTime
        create this with
          sanctionedHashes = sanctionedHashes ++ [partyHash]
          lastUpdated      = now

    -- ✅ FIXED: Same fix — removed `archive self`.
    choice RemoveSanctionedEntity : ContractId SanctionsRegistry
      with partyHash : Text
      controller compliance
      do
        now <- getTime
        create this with
          sanctionedHashes = filter (/= partyHash) sanctionedHashes
          lastUpdated      = now

    -- Check if a party is sanctioned (nonconsuming — compliance can check many parties)
    nonconsuming choice CheckSanctioned : Bool
      with partyHash : Text
      controller compliance
      do
        return (partyHash `elem` sanctionedHashes)

    -- Issue a clearance certificate after off-ledger check passes.
    -- settlementWindow: how long (seconds) the clearance is valid for.
    nonconsuming choice IssueClearance : ContractId SanctionsClearance
      with
        partyToCheck     : Party
        tradeId          : Text
        settlementWindow : Int   -- seconds; e.g. 3600 = 1 hour
      controller compliance
      do
        now <- getTime

        -- Compute hash and verify party is NOT sanctioned
        let partyHash    = sha256 (show partyToCheck)
        let isSanctioned = partyHash `elem` sanctionedHashes
        assertMsg ("Party is on sanctions list. Trade " <> tradeId <> " blocked.") (not isSanctioned)

        create SanctionsClearance with
          compliance   = compliance
          clearedParty = partyToCheck
          tradeId      = tradeId
          issuedAt     = now
          expiresAt    = addRelTime now (seconds settlementWindow)
          consumed     = False

-------------------------------
-- SANCTIONS CLEARANCE CERTIFICATE
-- Per-trade, short-lived, single-use proof that a party passed sanctions check.
-- DVP requires one for buyer and one for seller.
-------------------------------

template SanctionsClearance
  with
    compliance   : Party
    clearedParty : Party
    tradeId      : Text
    issuedAt     : Time
    expiresAt    : Time
    consumed     : Bool
  where
    signatory compliance
    observer clearedParty

    ensure not consumed  -- Active clearances only

    -- Called atomically during DVP settlement to consume the clearance

        pure ()

    -- Nonconsuming verify (for pre-flight checks without consuming)
    nonconsuming choice VerifyClearance : Bool
      controller compliance
      do
        now <- getTime
        return (now <= expiresAt && not consumed)

-- | Distribution - Dividend and yield distribution (REFACTORED FOR SECTION D.2)
-- ✅ SECTION D.2: Dividend "Pull" Model (Claim-Based)
-- ✅ FIXED: ClaimDividend is truly nonconsuming — no archive self, zero write-write contention.
--           Double-claim prevention uses a separate DividendClaimRecord keyed by (announcementId, holder).
-- ✅ FIXED: DividendAnnouncement now has an `eligibleHolders : [Party]` field.
--           DAML requires choice controllers to be stakeholders on the contract.
--           Previously `controller holder` where `holder` was a choice arg (not a stakeholder)
--           caused "contract not visible to reading parties" — Alice was not in the observer set.
--           `eligibleHolders` are added as `observer eligibleHolders` so any eligible holder
--           can exercise ClaimDividend. The issuer populates this from the snapshot off-ledger.
--
-- ORIGINAL BUG 1: ClaimDividend did `archive self` then `create this with claimedSoFar = ...`
-- This made it CONSUMING, causing O(N) write-write contention defeating the Pull model's purpose.
-- ORIGINAL BUG 2: `controller holder` where holder is a choice argument, not a stakeholder.
-- DAML rejects this: "contract not visible to reading parties".

module Distribution.ClaimBased where

import DA.Text (sha256)
import DA.Time
import DA.Map qualified as Map
import Portfolio (Holding_Impl(..))
import TokenStandard.Interfaces

-------------------------------
-- ✅ D.2: DIVIDEND ANNOUNCEMENT (Pull Model)
-------------------------------

template DividendAnnouncement
  with
    issuer               : Party
    compliance           : Party
    regulator            : Party
    assetId              : Text
    dividendLabel        : Text
    perUnitAmount        : Decimal
    snapshotTime         : Time
    expirationTime       : Time
    totalUnitsAtSnapshot : Decimal
    totalDividendPool    : Decimal
    announcementId       : Text
    -- ✅ FIXED: eligibleHolders are observers so they can exercise ClaimDividend.
    --           DAML's authorization model requires a choice's controller to be a
    --           stakeholder (signatory or observer) on the contract being exercised.
    --           Without this, any holder trying to call ClaimDividend gets:
    --           "contract not visible to reading parties".
    --           The issuer populates this list from the holding snapshot off-ledger
    --           before publishing the announcement. In production, this is the full
    --           list of token holders at snapshotTime.
    eligibleHolders      : [Party]
  where
    signatory issuer
    observer compliance, regulator, eligibleHolders

    ensure
      perUnitAmount > 0.0 &&
      totalUnitsAtSnapshot > 0.0 &&
      totalDividendPool == perUnitAmount * totalUnitsAtSnapshot &&
      expirationTime > snapshotTime &&
      announcementId /= ""

    -- ✅ Truly nonconsuming. All N holders can claim in parallel with zero contention.
    -- Double-claim prevention: DividendClaimRecord key (announcementId, holder)
    -- makes a second claim by the same holder an atomic duplicate-key failure.
    nonconsuming choice ClaimDividend : (ContractId DividendClaim, ContractId DividendClaimRecord)
      with
        holder           : Party
        holdingSnapshot  : Decimal
        proofOfOwnership : Text
      -- ✅ FIXED: controller is `holder` — valid because holder is in eligibleHolders,
      --           which is in the observer set, making holder a stakeholder.
      controller holder
      do
        now <- getTime
        assertMsg "Dividend claim window has expired" (now <= expirationTime)
        assertMsg "Snapshot balance must be positive"  (holdingSnapshot > 0.0)
        assertMsg "Holder is not in eligible holders list" (holder `elem` eligibleHolders)

        let holderShare = perUnitAmount * holdingSnapshot

        -- ✅ DividendClaimRecord key = (announcementId, holder)
        -- Second claim by same holder → duplicate key → atomic failure
        claimRecord <- create DividendClaimRecord with
          announcementId   = announcementId
          holder           = holder
          issuer           = issuer
          assetId          = assetId
          dividendLabel    = dividendLabel
          holdingSnapshot  = holdingSnapshot
          holderShare      = holderShare
          proofOfOwnership = proofOfOwnership
          claimedAt        = now

        claimCid <- create DividendClaim with
          issuer           = issuer
          holder           = holder
          assetId          = assetId
          dividendLabel    = dividendLabel
          amount           = holderShare
          holdingSnapshot  = holdingSnapshot
          claimedAt        = now
          proofOfOwnership = proofOfOwnership
          announcementId   = announcementId

        return (claimCid, claimRecord)

    choice CloseAnnouncement : ()
      controller issuer
      do
        now <- getTime
        assertMsg "Cannot close before expiration" (now > expirationTime)
        pure ()

    nonconsuming choice GetAnnouncementInfo : DividendAnnouncementInfo
      controller issuer, compliance, regulator
      do
        now <- getTime
        return DividendAnnouncementInfo with
          assetId           = assetId
          dividendLabel     = dividendLabel
          perUnitAmount     = perUnitAmount
          snapshotTime      = snapshotTime
          expirationTime    = expirationTime
          totalDividendPool = totalDividendPool
          isExpired         = now > expirationTime

data DividendAnnouncementInfo = DividendAnnouncementInfo
  with
    assetId           : Text
    dividendLabel     : Text
    perUnitAmount     : Decimal
    snapshotTime      : Time
    expirationTime    : Time
    totalDividendPool : Decimal
    isExpired         : Bool
  deriving (Eq, Show)

-------------------------------
-- ✅ DOUBLE-CLAIM PREVENTION RECORD
-- Key: (announcementId, holder). DAML enforces uniqueness per key globally.
-- A second ClaimDividend by the same holder for the same announcement fails atomically.
-- ✅ FIXED: `holder` must be a signatory (not just observer) because DAML requires all
--           key maintainers to also be signatories on the contract.
--           Previously `signatory issuer; observer holder` with `maintainer key._2` (holder)
--           caused: "some parties are maintainers but not signatories: 'Alice'".
--           Fix: `signatory issuer, holder` — holder co-signs their own claim record,
--           which is correct semantically (Alice should authorise her own claim).
--           The create is called from inside ClaimDividend which Alice controls, so
--           Alice's authority is already present in the transaction.
-------------------------------

template DividendClaimRecord
  with
    announcementId   : Text
    holder           : Party
    issuer           : Party
    assetId          : Text
    dividendLabel    : Text
    holdingSnapshot  : Decimal
    holderShare      : Decimal
    proofOfOwnership : Text
    claimedAt        : Time
  where
    signatory issuer, holder   -- ✅ FIXED: holder must be signatory to be a key maintainer

    ensure holderShare > 0.0

    key (announcementId, holder) : (Text, Party)
    maintainer key._2

-- | Finance.FractionalizationSafety - Safe Fractionalization Toggle
-- ✅ SECTION E.3: Fractionalization Toggle Safety
-- ✅ FIXED: CloseBuyBack return type includes ContractId
-- ✅ FIXED: AcceptDustBuyBack returns receiptCid not receipt
-- ✅ FIX AF-8: Removed explicit `archive self` from RecordDustSweepCompleted
--              (consuming choice already implicitly archives; double-archive caused the error)
-- ✅ FIX AF-11: Added `holder : Party` field to DustSweepBuyBack so the holder
--               is an observer and can see and exercise AcceptDustBuyBack
--
-- PROBLEM: Toggling fractionalized from true to false can break accounts
-- that hold fractional units (e.g., 0.5 shares). These become "dust" that
-- cannot be traded in integer-only mode.
--
-- SOLUTION: Either lock fractionalized as immutable after issuance, OR
-- require a "Dust Sweep" buy-back before allowing the toggle.

module Finance.FractionalizationSafety where

-- ✅ E.3: FRACTIONALIZATION POLICY
-------------------------------

data FractionalizationPolicy
  = ImmutableAfterIssuance      -- Once set, never changes
  | RequiresDustSweep           -- Can toggle after dust sweep
  | RequiresUnanimousConsent    -- All holders must approve
  deriving (Eq, Show)

template FractionalizationGovernance
  with
    issuer              : Party
    compliance          : Party
    assetId             : Text
    currentlyFractionalized : Bool
    policy              : FractionalizationPolicy
    issuanceDate        : Time
    canToggle           : Bool
  where
    signatory issuer
    observer compliance
    
    ensure
      -- If immutable, canToggle must be False
      (policy == ImmutableAfterIssuance) == (not canToggle)
    
    -- ✅ E.3: Attempt to toggle fractionalization
    choice RequestFractionalizationToggle : Either Text (ContractId FractionalizationToggleRequest)
      with
        newValue : Bool
        reason   : Text
      controller issuer
      do
        now <- getTime
        
        -- Check if toggle is allowed by policy
        case policy of
          ImmutableAfterIssuance -> do
            return (Left "Fractionalization is immutable after issuance per asset policy")
          
          RequiresDustSweep -> do
            if newValue == currentlyFractionalized then
              return (Left "New value same as current value")
            else do
              -- Create request that requires dust sweep
              req <- create FractionalizationToggleRequest with
                issuer = issuer
                compliance = compliance
                assetId = assetId
                currentValue = currentlyFractionalized
                newValue = newValue
                reason = reason
                requestedAt = now
                requiresDustSweep = not newValue  -- Only when going to integer-only
                dustSweepCompleted = False
                holderApprovalsRequired = False
                approvals = []
              return (Right req)
          
          RequiresUnanimousConsent -> do
            if newValue == currentlyFractionalized then
              return (Left "New value same as current value")
            else do
              req <- create FractionalizationToggleRequest with
                issuer = issuer
                compliance = compliance
                assetId = assetId
                currentValue = currentlyFractionalized
                newValue = newValue
                reason = reason
                requestedAt = now
                requiresDustSweep = not newValue
                dustSweepCompleted = False
                holderApprovalsRequired = True
                approvals = []
              return (Right req)

template FractionalizationToggleRequest

    compliance              : Party
    assetId                 : Text
    currentValue            : Bool
    newValue                : Bool
    reason                  : Text
    requestedAt             : Time
    requiresDustSweep       : Bool
    dustSweepCompleted      : Bool
    holderApprovalsRequired : Bool
    approvals               : [Party]  -- Holders who approved
  where
    signatory issuer
    observer compliance
    
    -- ✅ FIX AF-8: Removed explicit `archive self`.
    -- This is a consuming choice — DAML already implicitly archives.
    -- The explicit `archive self` caused a double-archive error.
    choice RecordDustSweepCompleted : ContractId FractionalizationToggleRequest
      with
        dustSweepProof : Text  -- ContractId of DustSweepCompletion
      controller compliance
      do
        assertMsg "Dust sweep not required" requiresDustSweep
        create this with dustSweepCompleted = True

    choice HolderApproval : ContractId FractionalizationToggleRequest
      with
        approver : Party
      controller approver
      do
        assertMsg "Holder approvals not required" holderApprovalsRequired
        assertMsg "Already approved" (not (approver `elem` approvals))
        
        archive self
        create this with approvals = approvals ++ [approver]
    
    -- ✅ E.3: Execute toggle (after all requirements met)
    choice ExecuteToggle : ContractId RWAInstrument

          pure ()

          assertMsg "Not all holders approved" (length approvals == length allHolders)
        else
          pure ()
        
        -- Fetch and update instrument
        instrument <- fetch instrumentCid
        archive instrumentCid
        
        -- Create updated instrument
        -- Note: In reality, you'd update the actual RWAInstrument template

        create instrument  -- Would be: create instrument { fractionalized = newValue }

-- Detect holdings with fractional amounts
template DustDetector
  with
    issuer     : Party
    compliance : Party
    assetId    : Text
  where
    signatory issuer
    observer compliance
    
    -- ✅ E.3: Scan for dust holdings
    nonconsuming choice DetectDustHoldings : DustReport
      with
        holdings : [(ContractId Holding, Decimal)]  -- (holdingCid, amount)
      controller issuer
      do
        -- Identify holdings with fractional amounts
        let dustHoldings = filter (\(_, amt) -> hasFractionalPart amt) holdings
        
        let totalDust = sum (map snd dustHoldings)
        let affectedHolders = length dustHoldings
        
        return DustReport with
          assetId = assetId
          totalDustHoldings = affectedHolders
          totalDustAmount = totalDust
          dustHoldings = map fst dustHoldings
          requiresSweep = affectedHolders > 0

data DustReport = DustReport
  with
    assetId           : Text
    totalDustHoldings : Int
    totalDustAmount   : Decimal
    dustHoldings      : [ContractId Holding]
    requiresSweep     : Bool
  deriving (Eq, Show)

-- Check if number has fractional part
hasFractionalPart : Decimal -> Bool
hasFractionalPart x = x /= intToDecimal (truncate x)

-- ✅ E.3: DUST SWEEP BUY-BACK
-- ✅ FIX AF-11: Added `holder : Party` field so the holder is an observer
--               and can see and exercise AcceptDustBuyBack on their contract.
--               Create one DustSweepBuyBack per holder (one-per-holder model).
-------------------------------

-- Issuer buys back fractional holdings for a specific holder
template DustSweepBuyBack
  with
    issuer        : Party
    compliance    : Party
    buyBackHolder : Party   -- ✅ FIX AF-11: per-holder field (renamed to avoid ambiguity)
    assetId       : Text
    buyBackPrice  : Decimal  -- Price per unit for dust buy-back
    deadline      : Time
  where
    signatory issuer
    observer compliance, buyBackHolder  -- ✅ FIX AF-11: holder can see their own buy-back offer
    
    ensure buyBackPrice > 0.0
    
    -- ✅ E.3: Holder accepts buy-back of dust
    -- ✅ FIXED: Returns receiptCid not receipt
    choice AcceptDustBuyBack : (ContractId Holding, ContractId DustBuyBackReceipt)
      with
        dustHoldingCid  : ContractId Holding
      controller buyBackHolder
      do
        now <- getTime
        assertMsg "Buy-back deadline passed" (now <= deadline)
        
        -- Fetch dust holding
        dustHolding <- fetch dustHoldingCid
        
        let amount = getAmount_H dustHolding
        assertMsg "Not a dust holding" (hasFractionalPart amount)
        assertMsg "Wrong asset" ((getInstrument_H dustHolding).id == assetId)
        
        -- Calculate payment
        let payment = amount * buyBackPrice
        
        -- Archive dust holding
        archive dustHoldingCid
        
        -- Create cash payment holding
        cashHolding <- create Holding_Impl with
          owner = buyBackHolder
          assetId = "USD"  -- Or whatever payment currency
          quantity = payment
          issuer = issuer
          compliance = compliance
          locked = False
          lockContext = None
          metadata = Map.fromList
            [ ("dust-buyback", "true")
            , ("original-asset", assetId)
            , ("dust-amount", show amount)
            , ("buyback-price", show buyBackPrice)
            ]
          observers   = [] 
        
        -- Create receipt
        receiptCid <- create DustBuyBackReceipt with
          issuer = issuer
          recipient = buyBackHolder
          assetId = assetId
          dustAmount = amount
          payment = payment
          completedAt = now
        
        return (toInterfaceContractId cashHolding, receiptCid)
    
    -- ✅ E.3: Close buy-back program
    -- ✅ FIXED: Return type is ContractId DustSweepCompletion
    choice CloseBuyBack : ContractId DustSweepCompletion
      with
        totalHoldersBoughtBack : Int
        totalDustSwept         : Decimal
      controller issuer
      do
        now <- getTime
        assertMsg "Cannot close before deadline" (now > deadline)
        
        create DustSweepCompletion with
          issuer = issuer
          compliance = compliance
          assetId = assetId
          completedAt = now
          holdersBoughtBack = totalHoldersBoughtBack
          totalDustSwept = totalDustSwept

template DustSweepCompletion
  with
    issuer            : Party
    compliance        : Party
    assetId           : Text
    completedAt       : Time
    holdersBoughtBack : Int
    totalDustSwept    : Decimal
  where
    signatory issuer
    observer compliance
    
    -- ✅ E.3: Verify no dust remains
    nonconsuming choice VerifyNoDustRemains : Bool
      with
        remainingHoldings : [(ContractId Holding, Decimal)]
      controller compliance
      do
        let dustRemains = any (\(_, amt) -> hasFractionalPart amt) remainingHoldings
        return (not dustRemains)

-------------------------------
-- ✅ E.3: AUTOMATIC DUST SWEEP COORDINATOR
-- ✅ NOTE: InitiateSweep now creates per-holder DustSweepBuyBack contracts.
--          dustHoldings is now [(Party, ContractId Holding)] to carry holder identity.
-------------------------------

-- Coordinates entire dust sweep process
template DustSweepCoordinator
  with
    coordinator   : Party
    compliance    : Party
    assetId       : Text
    buyBackPrice  : Decimal
    deadline      : Time
    dustHoldings  : [(Party, ContractId Holding)]  -- ✅ FIX AF-11: (holder, holdingCid) pairs
  where
    signatory coordinator
    observer compliance
    
    -- ✅ E.3: Initiate sweep — one DustSweepBuyBack per holder
    nonconsuming choice InitiateSweep : [ContractId DustSweepBuyBack]
      controller coordinator
      do
        forA dustHoldings $ \(holderParty, _) ->
          create DustSweepBuyBack with
            issuer = coordinator
            compliance = compliance
            buyBackHolder = holderParty
            assetId = assetId
            buyBackPrice = buyBackPrice
            deadline = deadline
    
    -- ✅ E.3: Check sweep progress
    nonconsuming choice CheckSweepProgress : SweepProgress
      with
        completedBuyBacks : Int
      controller coordinator
      do
        let totalDust = length dustHoldings
        let percentComplete = if totalDust == 0
                             then 100.0
                             else (intToDecimal completedBuyBacks / intToDecimal totalDust) * 100.0
        
        return SweepProgress with
          totalDustHoldings = totalDust
          completedBuyBacks = completedBuyBacks
          percentComplete = percentComplete
          remainingDust = totalDust - completedBuyBacks

-- Whitelist of acceptable collateral assets
template CollateralRegistry
  with
    registry       : Party
    compliance     : Party
    acceptedAssets : [(InstrumentId, CollateralTerms)]
  where
    signatory registry
    observer compliance
    
    -- ✅ F.4: Check if asset is accepted as collateral
    nonconsuming choice IsAcceptedCollateral : Bool
      with checkInstrument : InstrumentId
      controller registry
      do
        let accepted = any (\(inst, _) -> inst == checkInstrument) acceptedAssets
        return accepted
    
    -- ✅ F.4: Get collateral terms for asset
    nonconsuming choice GetCollateralTerms : Optional CollateralTerms
      with checkInstrument : InstrumentId
      controller registry
      do
        let result = filter (\(inst, _) -> inst == checkInstrument) acceptedAssets
        if null result then
          return None
        else
          return (Some (snd (head result)))
    
    -- ✅ FIXED: Removed explicit `archive self` - consuming choice archives automatically
    choice AddCollateralAsset : ContractId CollateralRegistry
      with
        newInstrument : InstrumentId
        terms         : CollateralTerms
      controller registry
      do
        now <- getTime
        create this with
          acceptedAssets = acceptedAssets ++ [(newInstrument, terms)]
    
    -- ✅ FIXED: Removed explicit `archive self` - consuming choice archives automatically
    choice RemoveCollateralAsset : ContractId CollateralRegistry
      with
        instrumentToRemove : InstrumentId
      controller registry
      do
        let filtered = filter (\(inst, _) -> inst /= instrumentToRemove) acceptedAssets
        create this with acceptedAssets = filtered

data CollateralTerms = CollateralTerms

    liquidationLTV   : Decimal  -- Liquidation threshold (e.g., 0.80 = 80%)
    minimumValueUSD  : Decimal  -- Minimum collateral value
    haircut          : Decimal  -- Discount applied to value (e.g., 0.10 = 10% haircut)
  deriving (Eq, Show)

template CrossRegistryLoanRequest
  with
    borrower             : Party
    poolOperator         : Party
    lendingAsset         : InstrumentId  -- What they want to borrow
    collateralAsset      : InstrumentId  -- ✅ F.4: Can be ANY CIP-0056 asset
    collateralCid        : ContractId Holding
    requestedAmount      : Decimal
    collateralAmount     : Decimal
    durationDays         : Int
    collateralRegistryCid: ContractId CollateralRegistry
    createdAt            : Time
    feeStructure         : FeeStructure  -- ✅ F.5: Explicit fees
  where
    signatory borrower
    observer poolOperator
    
    -- ✅ F.4 + F.5: Approve with cross-registry collateral + fees
    choice ApproveCrossRegistryLoan : (ContractId CrossRegistryLoan, Decimal)
      with
        poolCid                 : ContractId LendingPool
        collateralInstrumentCid : ContractId RWAInstrument  -- ✅ ADDED: Use instrument for price
      controller poolOperator
      do
        now <- getTime
        
        -- ✅ F.4: Verify collateral is accepted
        collateralRegistry <- fetch collateralRegistryCid
        isAccepted <- exercise collateralRegistryCid IsAcceptedCollateral with
          checkInstrument = collateralAsset
        assertMsg "Collateral asset not accepted" isAccepted
        
        -- Get collateral terms
        termsOpt : Optional CollateralTerms <- exercise collateralRegistryCid GetCollateralTerms with
          checkInstrument = collateralAsset
        
        -- Extract terms or fail
        terms <- case termsOpt of
          Some t -> return t
          None -> error "Collateral terms not found"
        
        -- ✅ UPDATED: Fetch instrument and use intrinsic price
        collateralInstrument <- fetch collateralInstrumentCid
        let price = collateralInstrument.pricePerUnit
        
        -- ✅ F.4: Calculate collateral value with haircut
        let collateralValueUSD = collateralAmount * price * (1.0 - terms.haircut)
        assertMsg "Collateral below minimum" (collateralValueUSD >= terms.minimumValueUSD)
        
        -- ✅ F.4: Verify LTV ratio
        let ltv = requestedAmount / collateralValueUSD
        assertMsg ("LTV too high. Max: " <> show terms.maxLTV) (ltv <= terms.maxLTV)
        
        -- Verify collateral holding
        collateral <- fetch collateralCid
        assertMsg "Collateral mismatch" (getInstrument_H collateral == collateralAsset)
        assertMsg "Insufficient collateral" (getAmount_H collateral >= collateralAmount)
        assertMsg "Collateral is locked" (not (isLocked collateral))
        
        -- ✅ F.5: Calculate origination fee
        let originationFee = calculateFee requestedAmount feeStructure.originationFeeBps
        let netLoanAmount = requestedAmount - originationFee
        
        -- Archive collateral
        archive collateralCid
        
        -- Create locked collateral holding
        lockedCollateral <- create Holding_Impl with
          owner = borrower
          assetId = collateralAsset.id
          quantity = collateralAmount
          issuer = collateralAsset.issuer
          compliance = poolOperator
          locked = True
          lockContext = Some ("Loan collateral: " <> show requestedAmount)
          metadata = Map.fromList
            [ ("loan-asset", lendingAsset.id)
            , ("loan-amount", show requestedAmount)
            , ("collateral-value-usd", show collateralValueUSD)
            , ("ltv-ratio", show ltv)
            ]
          observers   = [] 
        
        -- Return change if any
        if getAmount_H collateral > collateralAmount then do
          _ <- create Holding_Impl with
            owner = borrower
            assetId = collateralAsset.id
            quantity = getAmount_H collateral - collateralAmount
            issuer = collateralAsset.issuer
            compliance = poolOperator
            locked = False
            lockContext = None
            metadata = Map.empty
            observers   = [] 
          pure ()
        else pure ()
        
        -- Create loan principal holding for borrower
        loanHolding <- create Holding_Impl with
          owner = borrower
          assetId = lendingAsset.id
          quantity = netLoanAmount  -- ✅ F.5: After fee deduction
          issuer = poolOperator
          compliance = poolOperator
          locked = False
          lockContext = None
          metadata = Map.fromList
            [ ("loan-type", "cross-registry")
            , ("origination-fee", show originationFee)
            , ("gross-loan", show requestedAmount)
            ]
          observers   = [] 
        
        -- Create origination fee holding for pool operator
        feeHolding <- create Holding_Impl with
          owner = poolOperator
          assetId = lendingAsset.id
          quantity = originationFee
          issuer = poolOperator
          compliance = poolOperator
          locked = False
          lockContext = None
          metadata = Map.fromList
            [ ("fee-type", "origination")
            , ("borrower", show borrower)
            , ("loan-amount", show requestedAmount)
            ]
          observers   = [] 
        
        -- Create loan contract
        let maturityDate = addRelTime now (days durationDays)
        loan <- create CrossRegistryLoan with
          borrower = borrower
          lender = poolOperator
          lendingAsset = lendingAsset
          collateralAsset = collateralAsset
          principal = requestedAmount
          netPrincipal = netLoanAmount
          originationFee = originationFee
          collateralAmount = collateralAmount
          collateralValueUSD = collateralValueUSD
          lockedCollateralCid = toInterfaceContractId lockedCollateral
          originationDate = now
          maturityDate = maturityDate
          ltv = ltv
          liquidationLTV = terms.liquidationLTV
          feeStructure = feeStructure
          status = Active
        
        return (loan, originationFee)

-------------------------------

template CrossRegistryLoan
  with
    borrower            : Party
    lender              : Party
    lendingAsset        : InstrumentId
    collateralAsset     : InstrumentId  -- ✅ F.4: Cross-registry
    principal           : Decimal
    netPrincipal        : Decimal       -- ✅ F.5: After fees
    originationFee      : Decimal       -- ✅ F.5: Fee collected
    collateralAmount    : Decimal
    collateralValueUSD  : Decimal
    lockedCollateralCid : ContractId Holding
    originationDate     : Time
    maturityDate        : Time
    ltv                 : Decimal
    liquidationLTV      : Decimal
    feeStructure        : FeeStructure
    status              : LoanStatus
  where
    signatory borrower, lender
    
    -- ✅ F.5: Repay with management fee
    choice RepayLoanWithFees : (ContractId Holding, ContractId Holding, Decimal, Decimal)
      with
        repaymentHoldingCid : ContractId Holding
      controller borrower
      do
        now <- getTime
        
        -- Calculate interest
        let daysPassed = calculateDaysBetween originationDate now
        let interestAmount = (principal * 0.10 * daysPassed) / 365.0  -- 10% annual rate
        
        -- ✅ F.5: Calculate pro-rated management fee
        let managementFee = calculateProRataFee principal feeStructure.managementFeeBps (truncate daysPassed)
        
        -- Total repayment including fees
        let totalRepayment = principal + interestAmount + managementFee
        
        -- Verify repayment
        repayment <- fetch repaymentHoldingCid
        assertMsg "Insufficient repayment" (getAmount_H repayment >= totalRepayment)
        
        -- Archive repayment
        archive repaymentHoldingCid
        
        -- Return change if overpaid
        if getAmount_H repayment > totalRepayment then do
          _ <- create Holding_Impl with
            owner = borrower
            assetId = (getInstrument_H repayment).id
            quantity = getAmount_H repayment - totalRepayment
            issuer = lender
            compliance = lender
            locked = False
            lockContext = None
            metadata = Map.empty
            observers   = [] 
          pure ()
        else pure ()
        
        -- Create lender payment holding (principal + interest)
        lenderPayment <- create Holding_Impl with
          owner = lender
          assetId = lendingAsset.id
          quantity = principal + interestAmount
          issuer = lender
          compliance = lender
          locked = False
          lockContext = None
          metadata = Map.fromList
            [ ("loan-repayment", "true")
            , ("principal", show principal)
            , ("interest", show interestAmount)
            ]
          observers   = [] 
        
        -- ✅ F.5: Create management fee holding
        managementFeeHolding <- create Holding_Impl with
          owner = lender
          assetId = lendingAsset.id
          quantity = managementFee
          issuer = lender
          compliance = lender
          locked = False
          lockContext = None
          metadata = Map.fromList
            [ ("fee-type", "management")
            , ("borrower", show borrower)
            , ("days-charged", show daysPassed)
            ]
          observers   = [] 
        
        -- Unlock collateral
        lockedCollateral <- fetch lockedCollateralCid
        archive lockedCollateralCid
        
        unlockedCollateral <- create Holding_Impl with
          owner = borrower
          assetId = collateralAsset.id
          quantity = collateralAmount
          issuer = collateralAsset.issuer
          compliance = lender
          locked = False
          lockContext = None
          metadata = Map.fromList
            [ ("loan-repaid", "true")
            , ("repayment-date", show now)
            ]
          observers   = [] 
        
        return (
          toInterfaceContractId lenderPayment,
          toInterfaceContractId unlockedCollateral,
          interestAmount,
          managementFee
          )
    
    -- ✅ F.4: Liquidate with cross-registry collateral
    choice LiquidateCrossRegistryLoan : (ContractId Holding, Decimal)
      with collateralInstrumentCid : ContractId RWAInstrument  -- ✅ UPDATED: Use instrument for price
      controller lender
      do
        now <- getTime
        
        -- ✅ UPDATED: Fetch instrument and use intrinsic price
        collateralInstrument <- fetch collateralInstrumentCid
        let price = collateralInstrument.pricePerUnit
        
        let currentCollateralValueUSD = collateralAmount * price
        let currentLTV = principal / currentCollateralValueUSD
        
        -- Check if liquidation threshold breached
        assertMsg "LTV not breached" (currentLTV >= liquidationLTV)
        
        -- Transfer locked collateral to lender
        lockedCollateral <- fetch lockedCollateralCid
        archive lockedCollateralCid
        
        liquidatedCollateral <- create Holding_Impl with
          owner = lender
          assetId = collateralAsset.id
          quantity = collateralAmount
          issuer = collateralAsset.issuer
          compliance = lender
          locked = False
          lockContext = None
          metadata = Map.fromList
            [ ("liquidation", "true")
            , ("liquidation-ltv", show currentLTV)
            , ("liquidation-date", show now)
            , ("borrower", show borrower)
            ]
          observers   = [] 
        
        return (toInterfaceContractId liquidatedCollateral, currentLTV)

data TradeWithFeesResult = TradeWithFeesResult
  with
    buyerHolding   : ContractId Holding
    sellerHolding  : ContractId Holding
    platformFee    : Decimal
    feeHolding     : Optional (ContractId Holding)  -- ✅ CHANGED: Optional for zero-fee case
  deriving (Eq, Show)

template TradeAgreementWithFees
  with
    buyer          : Party
    seller         : Party
    issuer         : Party
    assetId        : Text
    quantity       : Decimal
    pricePerUnit   : Decimal
    feeStructure   : FeeStructure  -- ✅ F.5: Explicit fees
    platformParty  : Party         -- ✅ F.5: Fee recipient
  where
    signatory buyer, seller
    observer issuer, platformParty
    
    -- ✅ F.5: Execute with platform fees
    -- ✅ FIXED: Controller includes platformParty for authorization
    choice ExecuteTradeWithFees : TradeWithFeesResult
      with
        buyerPaymentCid : ContractId Holding
        sellerAssetCid  : ContractId Holding
      controller issuer, platformParty  -- ✅ CHANGED: Added platformParty
      do
        -- Verify holdings
        buyerPayment <- fetch buyerPaymentCid
        sellerAsset <- fetch sellerAssetCid
        
        let grossPrice = quantity * pricePerUnit
        
        -- ✅ F.5: Calculate trading fee
        let tradingFee = calculateFee grossPrice feeStructure.tradingFeeBps
        let netPrice = grossPrice - tradingFee
        
        assertMsg "Insufficient payment" (getAmount_H buyerPayment >= grossPrice)
        assertMsg "Insufficient assets" (getAmount_H sellerAsset >= quantity)
        
        -- Archive old holdings
        archive buyerPaymentCid
        archive sellerAssetCid
        
        now <- getTime
        
        -- Buyer receives asset
        buyerAsset <- create Holding_Impl with
          owner = buyer
          assetId = assetId
          quantity = quantity
          issuer = issuer
          compliance = issuer
          locked = False
          lockContext = None
          metadata = Map.fromList [("trade-fee-paid", show tradingFee)]
          observers   = [] 
        
        -- Seller receives net payment (after fees)
        sellerPayment <- create Holding_Impl with
          owner = seller
          assetId = (getInstrument_H buyerPayment).id
          quantity = netPrice
          issuer = issuer
          compliance = issuer
          locked = False
          lockContext = None
          metadata = Map.fromList [("net-after-fees", show netPrice)]
          observers   = [] 
        
        -- ✅ F.5: Platform receives trading fee (only if non-zero)
        -- ✅ FIXED: Don't create holding with 0.0 quantity
        feeHoldingOpt <- if tradingFee > 0.0 then do
          platformFeeHolding <- create Holding_Impl with
            owner = platformParty
            assetId = (getInstrument_H buyerPayment).id
            quantity = tradingFee
            issuer = issuer
            compliance = issuer
            locked = False
            lockContext = None
            metadata = Map.fromList
              [ ("fee-type", "trading")
              , ("trade-value", show grossPrice)
              , ("buyer", show buyer)
              , ("seller", show seller)
              ]
            observers   = [] 
          return (Some (toInterfaceContractId platformFeeHolding))
        else
          return None
        
        return TradeWithFeesResult with
          buyerHolding = toInterfaceContractId buyerAsset
          sellerHolding = toInterfaceContractId sellerPayment
          platformFee = tradingFee
          feeHolding = feeHoldingOpt  -- ✅ CHANGED: Now Optional

  let diffMicro = convertRelTimeToMicroseconds (subTime end start)
  in intToDecimal diffMicro / 1000000.0 / 86400.0

template DepositRequest
  with
    lender       : Party
    poolOperator : Party
    assetId      : Text
    amount       : Decimal
  where
    signatory lender
    observer poolOperator

    -- ✅ FIXED: Creates DepositEffect instead of direct pool update
    choice AcceptDeposit : (ContractId DepositEffect, ContractId LenderShare)
      with
        poolCid : ContractId LendingPool
      controller poolOperator
      do
        pool <- fetch poolCid
        assertMsg "Pool is Closed" (pool.status == Open)
        
        now <- getTime
        
        -- ✅ FIXED: Create deposit effect (doesn't consume pool)
        effectCid <- create DepositEffect with
          poolOperator = poolOperator
          user = lender
          poolCid = poolCid
          assetId = assetId
          amount = amount
          timestamp = now
          effectId = "DEPOSIT-" <> show lender <> "-" <> show now
          poolSequence = pool.sequenceNumber
        
        -- Create lender share (estimated - actual minted when effect is applied)
        let totalAssets = pool.totalLiquidity + pool.totalBorrowed
        let estimatedShares = if pool.totalShares == 0.0
                             then amount
                             else (amount * pool.totalShares) / totalAssets
        
        shareCid <- create LenderShare with
          lender = lender
          poolOperator = poolOperator
          assetId = assetId
          shareAmount = estimatedShares
        
        return (effectCid, shareCid)

    choice CancelDepositRequest : ()
      controller lender
      do pure ()

template WithdrawalRequest
  with
    lender       : Party
    poolOperator : Party
    assetId      : Text
    shares       : Decimal
  where

    observer poolOperator

    -- ✅ FIXED: Creates WithdrawalEffect and returns holding with compliance field
    choice ApproveWithdrawal : (ContractId WithdrawalEffect, ContractId Holding)
      with
        poolCid            : ContractId LendingPool
        existingHoldingCid : Optional (ContractId Holding)
      controller poolOperator, lender
      do
        pool <- fetch poolCid
        now <- getTime
        
        -- Calculate estimated payout
        let totalAssets = pool.totalLiquidity + pool.totalBorrowed
        let estimatedPayout = (shares * totalAssets) / pool.totalShares
        
        -- ✅ FIXED: Create withdrawal effect
        effectCid <- create WithdrawalEffect with
          poolOperator = poolOperator
          user = lender
          poolCid = poolCid
          assetId = assetId
          shares = shares
          timestamp = now
          effectId = "WITHDRAW-" <> show lender <> "-" <> show now
          poolSequence = pool.sequenceNumber
        
        -- ✅ FIXED: Create or merge holding with compliance field
        holdingCid <- case existingHoldingCid of
          Some allocCid -> do
            alloc <- fetch allocCid
            archive allocCid
            newHolding <- create Holding_Impl with
              owner = lender
              assetId = assetId
              quantity = getAmount_H alloc + estimatedPayout
              issuer = poolOperator
              compliance = pool.complianceParty  -- ✅ FIXED: Added compliance field
              locked = False
              lockContext = None
              metadata = emptyMetadata
              observers = []
            return (toInterfaceContractId newHolding)
          None -> do
            newHolding <- create Holding_Impl with
              owner = lender
              assetId = assetId
              quantity = estimatedPayout
              issuer = poolOperator
              compliance = pool.complianceParty  -- ✅ FIXED: Added compliance field
              locked = False
              lockContext = None
              metadata = emptyMetadata
              observers = []
            return (toInterfaceContractId newHolding)
        
        return (effectCid, holdingCid)

    choice RejectWithdrawal : ContractId LenderShare
      controller poolOperator
      do 
        create LenderShare with
          lender = lender
          poolOperator = poolOperator
          assetId = assetId
          shareAmount = shares

          poolOperator = poolOperator
          assetId = assetId
          shareAmount = shares

-- | Lending.Loans - Loan request and management
 
-- ✅ ENHANCED: Added explicit synchronizer compatibility checks to ApproveLoan
--              Mining rounds check timing, but synchronizer checks verify all contracts
--              (borrower collateral, pool liquidity, loan) are on the same synchronizer.

module Lending.Loans where

import DA.Time

template LoanRequest
  with
    borrower          : Party
    poolOperator      : Party
    assetId           : Text
    requestedAmount   : Decimal
    collateralAssetId : Text
    collateralAmount  : Decimal
    collateralCid     : ContractId Holding
    durationDays      : Int
    createdAt         : Time
    complianceParty   : Party
    regulatorParty    : Party
    miningRoundCid    : Optional (ContractId MiningRoundReference)
  where
    signatory borrower
    observer poolOperator, complianceParty

    -- ✅ ENHANCED: Added synchronizer compatibility parameters
    choice ApproveLoan : (ContractId Loan, ContractId TransferInstruction, ContractId LendingRegulatorView)
      with
        poolCid               : ContractId LendingPool
        existingCashCid       : Optional (ContractId Holding)
        -- ✅ NEW: Synchronizer compatibility parameters
        borrowerCurrentSync   : Text  -- Current sync where borrower's collateral resides
        poolCurrentSync       : Text  -- Current sync where pool's liquidity resides
        targetSync            : Text  -- Target sync for the loan transaction
      controller poolOperator
      do
        -- ✅ NEW: Explicit synchronizer compatibility check
        -- Mining rounds check timing/validity, but this checks that all contracts
        -- involved in the loan are on the same synchronizer for atomic execution
        let needsBorrowerMigration = borrowerCurrentSync /= targetSync
        let needsPoolMigration = poolCurrentSync /= targetSync
        
        if needsBorrowerMigration || needsPoolMigration then do
          let borrowerMsg = if needsBorrowerMigration
                           then "Borrower's collateral on '" <> borrowerCurrentSync <> "' must migrate to '" <> targetSync <> "'"
                           else ""
          let poolMsg = if needsPoolMigration
                       then "Pool's liquidity on '" <> poolCurrentSync <> "' must migrate to '" <> targetSync <> "'"
                       else ""
          let separator = if needsBorrowerMigration && needsPoolMigration then "; " else ""
          let errorMsg = "Cannot approve loan - synchronizer migration required: " <> borrowerMsg <> separator <> poolMsg
          assertMsg errorMsg False
        else
          pure ()

        pool <- fetch poolCid
        assertMsg "Pool is not Open" (pool.status == Open)
        assertMsg "Insufficient Liquidity" (pool.totalLiquidity >= requestedAmount)

        case miningRoundCid of
          Some roundCid -> do
            -- ✅ FIXED: Pass caller = poolOperator
            isValid <- exercise roundCid IsRoundValid with caller = poolOperator
            assertMsg "Canton Coin mining round expired" isValid
            hasTime <- exercise roundCid HasTimeForExternalSigning with
              estimatedSigningDelaySeconds = 120
              caller = poolOperator
            assertMsg "Not enough time in mining round" hasTime
          None -> pure ()

        collateral <- fetch collateralCid
        assertMsg "Collateral mismatch" ((getInstrument_H collateral).id == collateralAssetId)
        assertMsg "Insufficient collateral" (getAmount_H collateral >= collateralAmount)

        now <- getTime

        collateralTransferCid <- create TransferInstruction_Impl with
          sender     = borrower
          receiver   = poolOperator
          instrument = InstrumentId with
            depository = poolOperator
            issuer     = poolOperator
            id         = collateralAssetId
            version    = "0"
          amount           = collateralAmount
          deadline         = addRelTime now (hours 1)
          status           = Pending
          metadata         = Map.fromList
            [ ("cantonsuite.io/transfer-type", "LOAN-COLLATERAL")
            , ("cantonsuite.io/loan-amount",    show requestedAmount)
            , ("cantonsuite.io/initiated-at",   show now)
            , ("cantonsuite.io/synchronizer",   targetSync)  -- ✅ NEW: Track sync
            ]
          senderHoldingCid = collateralCid

        archive collateralCid

        if getAmount_H collateral > collateralAmount then do
          _ <- create Holding_Impl with
            owner       = borrower
            assetId     = collateralAssetId
            quantity    = getAmount_H collateral - collateralAmount
            issuer      = poolOperator
            compliance  = pool.complianceParty
            locked      = False
            lockContext = None
            metadata    = emptyMetadata
            observers   = [] 
          pure ()
        else pure ()

        archive poolCid
        newPoolCid <- create pool with
          totalLiquidity = pool.totalLiquidity - requestedAmount
          totalBorrowed  = pool.totalBorrowed + requestedAmount
          sequenceNumber = pool.sequenceNumber + 1

        loanHoldingCid <- case existingCashCid of
          Some cid -> do
            c <- fetch cid
            archive cid
            create Holding_Impl with
              owner       = borrower
              assetId     = assetId
              quantity    = getAmount_H c + requestedAmount
              issuer      = poolOperator
              compliance  = pool.complianceParty
              locked      = False
              lockContext = None
              metadata    = emptyMetadata
              observers   = [] 
          None ->
            create Holding_Impl with
              owner       = borrower
              assetId     = assetId
              quantity    = requestedAmount
              issuer      = poolOperator
              compliance  = pool.complianceParty
              locked      = False
              lockContext = None
              metadata    = emptyMetadata
              observers   = [] 

        let maturityDate = addRelTime now (days durationDays)

        loanCid <- create Loan with
          borrower              = borrower
          lender                = poolOperator
          assetId               = assetId
          principal             = requestedAmount
          interestRate          = pool.interestRate
          collateralAssetId     = collateralAssetId
          collateralAmount      = collateralAmount
          collateralTransferCid = toInterfaceContractId collateralTransferCid
          originationDate       = now
          maturityDate          = maturityDate
          status                = Active
          regulatorParty        = regulatorParty
          complianceParty       = pool.complianceParty
          miningRoundCid        = miningRoundCid

        let borrowerHash = sha256 (show borrower)
        let loanIdHash   = sha256 (show loanCid <> show now)

        regulatorCid <- create LendingRegulatorView with
          regulator       = regulatorParty
          poolOperator    = poolOperator
          loanIdHash      = loanIdHash
          borrowerHash    = borrowerHash
          assetId         = assetId
          principal       = requestedAmount
          collateralRatio = (collateralAmount / requestedAmount) * 100.0
          interestRate    = pool.interestRate
          status          = "ACTIVE"
          originationDate = now
          maturityDate    = maturityDate
          eventType       = "LOAN_ORIGINATED"
          eventTimestamp  = now

        return (loanCid, toInterfaceContractId collateralTransferCid, regulatorCid)

    choice RejectLoan : ()
      controller poolOperator
      do pure ()

    choice CancelRequest : ()
      controller borrower
      do pure ()

template Loan
  with
    borrower              : Party
    lender                : Party
    assetId               : Text
    principal             : Decimal
    interestRate          : InterestRate
    collateralAssetId     : Text
    collateralAmount      : Decimal
    collateralTransferCid : ContractId TransferInstruction
    originationDate       : Time
    maturityDate          : Time
    status                : LoanStatus
    regulatorParty        : Party
    complianceParty       : Party
    miningRoundCid        : Optional (ContractId MiningRoundReference)
  where
    signatory borrower, lender
    observer complianceParty

    ensure principal > 0.0

    choice RepayLoan : (ContractId TransferInstruction, ContractId TransferInstruction, ContractId LendingPool, ContractId LendingRegulatorView)
      with
        repaymentHoldingCid   : ContractId Holding
        poolCid               : ContractId LendingPool
        existingCollateralCid : Optional (ContractId Holding)
      controller borrower
      do
        case miningRoundCid of
          Some roundCid -> do
            -- ✅ FIXED: Pass caller = borrower
            isValid <- exercise roundCid IsRoundValid with caller = borrower
            assertMsg "Canton Coin mining round expired" isValid
          None -> pure ()

        now <- getTime
        let daysPassed     = calculateDaysBetween originationDate now
        let interestAmount = principal * (interestRate / 100.0) * (daysPassed / 365.0)
        let totalRepayment = principal + interestAmount

        repayment <- fetch repaymentHoldingCid
        assertMsg ("Insufficient funds. Need " <> show totalRepayment) (getAmount_H repayment >= totalRepayment)

        repaymentTransferCid <- create TransferInstruction_Impl with
          sender     = borrower
          receiver   = lender
          instrument = InstrumentId with
            depository = lender
            issuer     = lender
            id         = assetId
            version    = "0"
          amount           = totalRepayment
          deadline         = addRelTime now (hours 1)
          status           = Pending
          metadata         = Map.fromList
            [ ("cantonsuite.io/transfer-type", "LOAN-REPAYMENT")
            , ("cantonsuite.io/principal",      show principal)
            , ("cantonsuite.io/interest",        show interestAmount)
            , ("cantonsuite.io/initiated-at",    show now)
            , (cip0056Reason, "Loan repayment with interest")
            ]
          senderHoldingCid = repaymentHoldingCid

        archive repaymentHoldingCid

        if getAmount_H repayment > totalRepayment then do
          _ <- create Holding_Impl with
            owner       = borrower
            assetId     = (getInstrument_H repayment).id
            quantity    = getAmount_H repayment - totalRepayment
            issuer      = lender
            compliance  = complianceParty
            locked      = False
            lockContext = None
            metadata    = emptyMetadata
            observers   = [] 
          pure ()
        else pure ()

        pool <- fetch poolCid
        archive poolCid
        newPoolCid <- create pool with
          totalBorrowed  = pool.totalBorrowed - principal
          totalLiquidity = pool.totalLiquidity + principal + interestAmount
          sequenceNumber = pool.sequenceNumber + 1

        let concreteCollateralTransferCid = coerceContractId @TransferInstruction @TransferInstruction_Impl collateralTransferCid
        concreteCollateralTransfer <- fetch concreteCollateralTransferCid
        let collateralHoldingCid = concreteCollateralTransfer.senderHoldingCid

        collateralReturnCid <- create TransferInstruction_Impl with
          sender     = lender
          receiver   = borrower
          instrument = InstrumentId with
            depository = lender
            issuer     = lender
            id         = collateralAssetId
            version    = "0"
          amount           = collateralAmount
          deadline         = addRelTime now (hours 1)
          status           = Pending
          metadata         = Map.fromList
            [ ("cantonsuite.io/transfer-type", "COLLATERAL-RETURN")
            , ("cantonsuite.io/loan-id",        show self)
            , ("cantonsuite.io/initiated-at",    show now)
            , (cip0056Reason, "Collateral release after loan repayment")
            ]
          senderHoldingCid = collateralHoldingCid

        let borrowerHash = sha256 (show borrower)
        let loanIdHash   = sha256 (show self <> show originationDate)

        regulatorCid <- create LendingRegulatorView with
          regulator       = regulatorParty
          poolOperator    = lender
          loanIdHash      = loanIdHash
          borrowerHash    = borrowerHash
          assetId         = assetId
          principal       = principal
          collateralRatio = (collateralAmount / principal) * 100.0
          interestRate    = interestRate
          status          = "REPAID"
          originationDate = originationDate
          maturityDate    = maturityDate
          eventType       = "LOAN_REPAID"
          eventTimestamp  = now

        return (toInterfaceContractId repaymentTransferCid, toInterfaceContractId collateralReturnCid, newPoolCid, regulatorCid)

    choice LiquidateCollateral : (ContractId TransferInstruction, ContractId LendingPool, ContractId LendingRegulatorView)
      with
        poolCid : ContractId LendingPool
      controller lender
      do
        now <- getTime
        assertMsg "Loan not yet matured" (now > maturityDate)

        -- ✅ FIXED: Check mining round if present (caller = lender)
        case miningRoundCid of
          Some roundCid -> do
            isValid <- exercise roundCid IsRoundValid with caller = lender
            assertMsg "Canton Coin mining round expired" isValid
            
            hasTime <- exercise roundCid HasTimeForExternalSigning with
              estimatedSigningDelaySeconds = 120
              caller = lender
            assertMsg "Not enough time in mining round for liquidation" hasTime
          None -> pure ()

        let concreteCollateralTransferCid = coerceContractId @TransferInstruction @TransferInstruction_Impl collateralTransferCid
        concreteCollateralTransfer <- fetch concreteCollateralTransferCid
        let collateralHoldingCid = concreteCollateralTransfer.senderHoldingCid

        liquidationTransferCid <- create TransferInstruction_Impl with
          sender     = borrower
          receiver   = lender
          instrument = InstrumentId with
            depository = lender
            issuer     = lender
            id         = collateralAssetId
            version    = "0"
          amount           = collateralAmount
          deadline         = addRelTime now (hours 1)
          status           = Executed
          metadata         = Map.fromList
            [ ("cantonsuite.io/transfer-type",    "COLLATERAL-LIQUIDATION")
            , ("cantonsuite.io/loan-id",           show self)
            , ("cantonsuite.io/defaulted-amount",  show principal)
            , ("cantonsuite.io/initiated-at",      show now)
            , (cip0056Reason, "Collateral liquidated due to loan default")
            ]
          
          
          senderHoldingCid = collateralHoldingCid

        pool <- fetch poolCid
        archive poolCid
        newPoolCid <- create pool with
          totalBorrowed  = pool.totalBorrowed - principal
          sequenceNumber = pool.sequenceNumber + 1

        let borrowerHash = sha256 (show borrower)
        let loanIdHash   = sha256 (show self <> show originationDate)

        regulatorCid <- create LendingRegulatorView with
          regulator       = regulatorParty
          poolOperator    = lender
          loanIdHash      = loanIdHash
          borrowerHash    = borrowerHash
          assetId         = assetId
          principal       = principal
          collateralRatio = (collateralAmount / principal) * 100.0
          interestRate    = interestRate
          status          = "LIQUIDATED"
          originationDate = originationDate
          maturityDate    = maturityDate
          eventType       = "LOAN_LIQUIDATED"
          eventTimestamp  = now

        return (toInterfaceContractId liquidationTransferCid, newPoolCid, regulatorCid)

    choice ExtendLoan : ContractId Loan
      with additionalDays : Int
      controller borrower, lender
      do
        create this with maturityDate = addRelTime maturityDate (days additionalDays)

    status          : Text

     then 0.0
     else intToDecimal diffMicro / 1000000.0 / 86400.0

  with

    totalBorrowed  : Decimal

    collateralRatio: Decimal

    -- Multiple users can query simultaneously
    nonconsuming choice GetSharePrice : Decimal
      controller poolOperator
      do
        let totalAssets = totalLiquidity + totalBorrowed
        if totalShares == 0.0 then return 1.0
        else return (totalAssets / totalShares)

        let totalAssets = totalLiquidity + totalBorrowed
        let payout = (shares * totalAssets) / totalShares
        assertMsg "Insufficient Liquidity in Pool" (totalLiquidity >= payout)

    choice UpdateInterestRate : ContractId LendingPool
      with newRate : InterestRate
      controller poolOperator
      do
        create this with
          interestRate = newRate
          sequenceNumber = sequenceNumber + 1
    
    choice InitiateClosure : ContractId LendingPool
      controller poolOperator
      do
        create this with
          status = Closing

-- Deposit Effect
template DepositEffect
  with
    poolOperator   : Party
    user           : Party
    poolCid        : ContractId LendingPool
    assetId        : Text
    amount         : Decimal
    timestamp      : Time
    effectId       : Text
    poolSequence   : Int  -- ✅ D.1: For optimistic concurrency check
  where
    signatory user
    observer poolOperator
    
    ensure amount > 0.0
    
    -- ✅ D.1: Operator applies effect in batch
    choice ApplyDepositEffect : (ContractId LendingPool, Decimal)
      controller poolOperator
      do
        pool <- fetch poolCid
        
        -- Verify pool hasn't changed (optimistic concurrency)
        assertMsg "Pool state changed, retry" (pool.sequenceNumber == poolSequence)
        assertMsg "Pool is Closed" (pool.status == Open)
        
        -- Calculate shares to mint
        let totalAssets = pool.totalLiquidity + pool.totalBorrowed
        let sharesToMint = if pool.totalShares == 0.0
                          then amount
                          else (amount * pool.totalShares) / totalAssets
        
        -- Archive old pool and create new one
        archive poolCid
        newPoolCid <- create pool with
          totalLiquidity = pool.totalLiquidity + amount
          totalShares = pool.totalShares + sharesToMint
          sequenceNumber = pool.sequenceNumber + 1
        
        return (newPoolCid, sharesToMint)
    
    choice CancelDepositEffect : ()
      controller user
      do pure ()

-------------------------------
-- ✅ D.1: BATCH EFFECT PROCESSOR
-- Operator can process multiple effects in one transaction
-------------------------------

template BatchEffectProcessor
  with
    poolOperator : Party
    poolCid      : ContractId LendingPool
  where
    signatory poolOperator
    
    -- ✅ Process multiple deposits atomically
    nonconsuming choice ProcessDepositBatch : ContractId LendingPool
      with
        depositEffectCids : [ContractId DepositEffect]
      controller poolOperator
      do
        -- Apply each effect sequentially
        foldlA (\currentPoolCid effectCid -> do
          (newPoolCid, _) <- exercise effectCid ApplyDepositEffect
          return newPoolCid
          ) poolCid depositEffectCids
    
    -- ✅ Process multiple withdrawals atomically
    nonconsuming choice ProcessWithdrawalBatch : ContractId LendingPool
      with
        withdrawalEffectCids : [ContractId WithdrawalEffect]
      controller poolOperator
      do
        foldlA (\currentPoolCid effectCid -> do
          (newPoolCid, _) <- exercise effectCid ApplyWithdrawalEffect
          return newPoolCid
          ) poolCid withdrawalEffectCids
    
    -- ✅ Process mixed effects in optimal order
    nonconsuming choice ProcessMixedBatch : ContractId LendingPool
      with
        depositCids    : [ContractId DepositEffect]
        withdrawalCids : [ContractId WithdrawalEffect]
        borrowCids     : [ContractId BorrowEffect]
        repaymentCids  : [ContractId RepaymentEffect]
      controller poolOperator
      do
        -- Process repayments first (add liquidity)
        poolAfterRepayments <- foldlA (\cid effectCid -> 
          exercise effectCid ApplyRepaymentEffect
          ) poolCid repaymentCids
        
        -- Then deposits (more liquidity)
        poolAfterDeposits <- foldlA (\cid effectCid -> do
          (newCid, _) <- exercise effectCid ApplyDepositEffect
          return newCid
          ) poolAfterRepayments depositCids
        
        -- Then borrows (reduce liquidity)
        poolAfterBorrows <- foldlA (\cid effectCid -> 
          exercise effectCid ApplyBorrowEffect
          ) poolAfterDeposits borrowCids
        
        -- Finally withdrawals (further reduce)
        foldlA (\cid effectCid -> do
          (newCid, _) <- exercise effectCid ApplyWithdrawalEffect
          return newCid
          ) poolAfterBorrows withdrawalCids

-------------------------------
-- ✅ FIXED: HELPER FUNCTION
-- Proper indentation and do syntax
-------------------------------

foldlA : (b -> a -> Update b) -> b -> [a] -> Update b
foldlA f acc [] = return acc
foldlA f acc (x :: xs) = do
  newAcc <- f acc x
  foldlA f newAcc xs

template SubscriptionRequest
  with
    issuer        : Party
    subscriber    : Party
    assetId       : Text
    -- ✅ FIXED: Changed from `ContractId RWAInstrument` to `Optional (ContractId RWAInstrument)`.
    --           The field is never used in any choice body. The old tests passed
    --           `instrumentCid = error "not used"` / `error "not used in accept"` which
    --           throws GeneralError at contract creation time (DAML evaluates record fields
    --           eagerly). `Optional` allows `None` to be passed safely.
    instrumentCid : Optional (ContractId RWAInstrument)
    quantity      : Decimal
    pricePerUnit  : Decimal
    expiresAt     : Time
    createdAt     : Time
    compliance    : Party
    offerType     : Text
  where
    signatory issuer
    observer subscriber, compliance

    -- ✅ FIXED: controller is `issuer, subscriber` — not just `subscriber`.
    --           The choice body fetches issuerAssetCid (owned by issuer, visible only to
    --           issuer/compliance). With only subscriber as controller, Alice's transaction
    --           cannot read the issuer's holding: "contract not visible to reading parties".
    --           Adding issuer as co-controller brings issuer's read authority in.
    --           Semantically correct: issuer must consent to releasing their asset.
    --           Test uses `submitMulti [issuer, alice]` for this choice.
    choice AcceptSubscription : (ContractId Holding, ContractId Holding)
      with
        paymentHoldingCid : ContractId Holding
        issuerAssetCid    : ContractId Holding
      controller issuer, subscriber
      do
        now <- getTime
        assertMsg "Subscription expired" (now <= expiresAt)

        payment     <- fetch paymentHoldingCid
        issuerAsset <- fetch issuerAssetCid

        let totalPrice = quantity * pricePerUnit
        assertMsg "Insufficient payment" (getAmount_H payment >= totalPrice)
        assertMsg "Insufficient assets"  (getAmount_H issuerAsset >= quantity)

        archive paymentHoldingCid
        archive issuerAssetCid

        subscriberAsset <- create Holding_Impl with
          owner       = subscriber
          assetId     = assetId
          quantity    = quantity
          issuer      = issuer
          compliance  = compliance
          locked      = False
          lockContext = None
          metadata    = Map.fromList
            [ ("subscription-type", "bilateral")
            , ("offer-type",        offerType)
            , ("settlement-time",   show now)
            ]
          observers   = [] 

        issuerPayment <- create Holding_Impl with
          owner       = issuer
          assetId     = (getInstrument_H payment).id
          quantity    = totalPrice
          issuer      = issuer
          compliance  = compliance
          locked      = False
          lockContext = None
          metadata    = Map.fromList
            [ ("subscription-payment", "true")
            , ("subscriber",           show subscriber)
            ]
          observers   = [] 

        if getAmount_H payment > totalPrice then do
          _ <- create Holding_Impl with
            owner       = subscriber
            assetId     = (getInstrument_H payment).id
            quantity    = getAmount_H payment - totalPrice
            issuer      = issuer
            compliance  = compliance
            locked      = False
            lockContext = None
            metadata    = Map.empty
            observers   = [] 
          pure ()
        else pure ()

        if getAmount_H issuerAsset > quantity then do
          _ <- create Holding_Impl with
            owner       = issuer
            assetId     = assetId
            quantity    = getAmount_H issuerAsset - quantity
            issuer      = issuer
            compliance  = compliance
            locked      = False
            lockContext = None
            metadata    = Map.empty
            observers   = [] 
          pure ()
        else pure ()

        return (toInterfaceContractId subscriberAsset, toInterfaceContractId issuerPayment)

    choice DeclineSubscription : ()
      controller subscriber
      do pure ()

template SynchronizerCompatibilityCheck
  with
    checker          : Party
    contracts        : [(ContractId Holding, Text)]
    settlementId     : Text
    requiredSync     : Text
  where
    signatory checker
    
    nonconsuming choice CheckCompatibility : CompatibilityResult
      controller checker
      do
        let incompatible = filter (\(_, sync) -> sync /= requiredSync) contracts
        
        if null incompatible then
          return CompatibilityResult with
            compatible = True
            incompatibleContracts = []
            recommendedAction = None
        else
          return CompatibilityResult with
            compatible = False
            incompatibleContracts = map fst incompatible
            recommendedAction = Some ("Migrate " <> show (length incompatible) <> " contracts to " <> requiredSync)

data CompatibilityResult = CompatibilityResult
  with
    compatible            : Bool
    incompatibleContracts : [ContractId Holding]
    recommendedAction     : Optional Text
  deriving (Eq, Show)

data MigrationStatus
  = MigrationPending
  | MigrationInProgress
  | MigrationCompleted
  | MigrationFailed
  deriving (Eq, Show)

template MigrationWorkflow
  with
    requester   : Party
    issuer      : Party
    holdingCid  : ContractId Holding
    currentSync : Text
    targetSync  : Text
    reason      : Text
    status      : MigrationStatus
    initiatedAt : Time
  where
    signatory issuer, requester
    
    choice LockForMigration : (ContractId Holding, ContractId MigrationWorkflow)
      controller issuer
      do
        assertMsg "Migration must be pending" (status == MigrationPending)
        
        holding <- fetch holdingCid
        assertMsg "Holding is already locked" (not (isLocked holding))
        
        archive holdingCid
        
        lockedHolding <- create Holding_Impl with
          owner = getAccount holding |> (.owner)
          assetId = (getInstrument_H holding).id
          quantity = getAmount_H holding
          issuer = issuer
          compliance = issuer
          locked = True
          lockContext = Some ("Migration to " <> targetSync)
          metadata = addMetadata "migration-in-progress" "true" (Map.empty)
          observers = []
        
      
        newWorkflow <- create this with
          status = MigrationInProgress
          holdingCid = toInterfaceContractId lockedHolding
        
        return (toInterfaceContractId lockedHolding, newWorkflow)
    
    choice CompleteMigration : (ContractId Holding, ContractId HoldingSynchronizerMetadata)
      with
        migrationProof : Text
      controller issuer
      do
        assertMsg "Migration must be in progress" (status == MigrationInProgress)
        
        now <- getTime
        
        holding <- fetch holdingCid
        assertMsg "Holding must be locked" (isLocked holding)
        
        archive holdingCid
        
        newHolding <- create Holding_Impl with
          owner = getAccount holding |> (.owner)
          assetId = (getInstrument_H holding).id
          quantity = getAmount_H holding
          issuer = issuer
          compliance = issuer
          locked = False
          lockContext = None
          metadata = Map.fromList
            [ ("migrated-from", currentSync)
            , ("migrated-to", targetSync)
            , ("migration-proof", migrationProof)
            , ("migration-completed-at", show now)
            ]
          observers = []
        
        syncMeta <- create HoldingSynchronizerMetadata with
          owner = getAccount holding |> (.owner)
          issuer = issuer
          assetId = (getInstrument_H holding).id
          holdingCid = toInterfaceContractId newHolding
          synchronizerInfo = SynchronizerInfo with
            synchronizerId = targetSync
            synchronizerUrl = getSyncUrl targetSync
            connectedAt = now
            lastVerified = now
          canMigrate = True
        
        
        _ <- create this with status = MigrationCompleted
        
        return (toInterfaceContractId newHolding, syncMeta)
    
    choice FailMigration : ContractId Holding
      with
        failureReason : Text
      controller issuer
      do
        assertMsg "Migration must be in progress" (status == MigrationInProgress)
        
        holding <- fetch holdingCid
        
        archive holdingCid
        
        restoredHolding <- create Holding_Impl with
          owner = getAccount holding |> (.owner)
          assetId = (getInstrument_H holding).id
          quantity = getAmount_H holding
          issuer = issuer
          compliance = issuer
          locked = False
          lockContext = None
          metadata = Map.fromList
            [ ("migration-failed", "true")
            , ("failure-reason", failureReason)
            ]
          observers = []
        
        
        _ <- create this with status = MigrationFailed
        
        return (toInterfaceContractId restoredHolding)

template PreSettlementMigrationCoordinator
  with
    coordinator      : Party
    settlementId     : Text
    targetSync       : Text
    buyerHoldings    : [ContractId Holding]
    sellerHoldings   : [ContractId Holding]
    deadline         : Time
  where
    signatory coordinator
    
    nonconsuming choice CheckMigrationNeeded : MigrationPlan
      with
        holdingSyncMeta : [(ContractId Holding, Text)]
      controller coordinator
      do
        let needMigration = filter (\(_, sync) -> sync /= targetSync) holdingSyncMeta
        
        if null needMigration then
          return MigrationPlan with
            migrationRequired = False
            holdingsToMigrate = []
            estimatedTime = 0
        else
          return MigrationPlan with
            migrationRequired = True
            holdingsToMigrate = map fst needMigration
            estimatedTime = length needMigration * 30
    
    choice InitiateBatchMigration : [ContractId SynchronizerMigrationRequest]
      with
        holdingSyncInfo : [(ContractId Holding, Text)]
      controller coordinator
      do
        now <- getTime
        
        forA (filter (\(_, sync) -> sync /= targetSync) holdingSyncInfo) $ 
          \(holdingCid, currentSync) -> do
            holding <- fetch holdingCid
            
            create SynchronizerMigrationRequest with
              requester = coordinator
              issuer = (getInstrument_H holding).issuer
              holdingCid = holdingCid
              currentSync = currentSync
              targetSync = targetSync
              reason = "Pre-settlement migration for " <> settlementId
              requestedAt = now
              deadline = deadline

data MigrationPlan = MigrationPlan
  with
    migrationRequired : Bool
    holdingsToMigrate : [ContractId Holding]
    estimatedTime     : Int
  deriving (Eq, Show)

template DVPSettlementWithMigration
  with
    settlementId     : Text
    settlementApp    : Party
    buyer            : Party
    seller           : Party
    buyerHoldingCid  : ContractId Holding
    sellerHoldingCid : ContractId Holding
    deadline         : Time
    targetSync       : Text
  where
    signatory settlementApp
    observer buyer, seller
    
    choice PrepareForSettlement : PreparationResult
      with
        buyerCurrentSync  : Text
        sellerCurrentSync : Text
      controller settlementApp
      do
        now <- getTime
        assertMsg "Settlement deadline passed" (now <= deadline)
        
        let buyerNeedsMigration = buyerCurrentSync /= targetSync
        let sellerNeedsMigration = sellerCurrentSync /= targetSync
        
        if not buyerNeedsMigration && not sellerNeedsMigration then
          return PreparationResult with
            ready = True
            buyerMigrationCid = None
            sellerMigrationCid = None
            estimatedWaitTime = 0
        else do
          buyerMigReq <- if buyerNeedsMigration then do
            req <- create SynchronizerMigrationRequest with
              requester = buyer
              issuer = settlementApp
              holdingCid = buyerHoldingCid
              currentSync = buyerCurrentSync
              targetSync = targetSync
              reason = "Pre-settlement migration"
              requestedAt = now
              deadline = deadline
            return (Some req)
          else
            return None
          
          sellerMigReq <- if sellerNeedsMigration then do
            req <- create SynchronizerMigrationRequest with
              requester = seller
              issuer = settlementApp
              holdingCid = sellerHoldingCid
              currentSync = sellerCurrentSync
              targetSync = targetSync
              reason = "Pre-settlement migration"
              requestedAt = now
              deadline = deadline
            return (Some req)
          else
            return None
          
          let waitTime = (if buyerNeedsMigration then 30 else 0) + 
                        (if sellerNeedsMigration then 30 else 0)
          
          return PreparationResult with
            ready = False
            buyerMigrationCid = buyerMigReq
            sellerMigrationCid = sellerMigReq
            estimatedWaitTime = waitTime

data PreparationResult = PreparationResult
  with
    ready              : Bool
    buyerMigrationCid  : Optional (ContractId SynchronizerMigrationRequest)
    sellerMigrationCid : Optional (ContractId SynchronizerMigrationRequest)
    estimatedWaitTime  : Int
  deriving (Eq, Show)

-- | Portfolio - CIP-0056 Compliant Holdings
--
-- ✅ SECTION A.1: Owner-only signatory on normal holdings (privacy)
-- ✅ SECTION A.3: Dual-auth Clawback via separate LockedHolding_Impl template
--
-- DESIGN: Two distinct templates encode two distinct legal states.
--
--   Holding_Impl        — normal state. signatory owner ONLY.
--                         Issuer and compliance are observers.
--                         CIP-0056 privacy fully preserved.
--
--   LockedHolding_Impl  — clawback-pending state. signatory owner, issuer, compliance.
--                         Elevated signatories reflect real-world legal reality.
--
-- GOVERNANCE: InitiateClawback requires BOTH issuer AND compliance as controllers.
-- No single party can unilaterally freeze assets. This is better than real-world
-- practice where a single compliance officer can often freeze unilaterally —
-- here we enforce dual-party initiation on-ledger. AuthorizeClawback then
-- serves as the formal on-record sign-off with the court order hash.

module Portfolio where

import TokenStandard.Interfaces
import DA.Map qualified as Map

----------------------------------------------------------------------
-- HOLDING_IMPL
-- Normal holding. Owner is sole signatory — CIP-0056 / Section A.1.
----------------------------------------------------------------------

template Holding_Impl
  with
    owner       : Party
    assetId     : Text
    quantity    : Decimal
    issuer      : Party
    compliance  : Party
    locked      : Bool
    lockContext : Optional Text
    metadata    : Metadata
    observers   : [Party]
  where
    signatory owner
    observer issuer, compliance, observers

    ensure quantity > 0.0

    interface instance Holding for Holding_Impl where
      view = HoldingView with
        instrument = InstrumentId with
          depository = issuer
          issuer     = issuer
          id         = assetId
          version    = "0"
        account = AccountId with
          custodian = issuer
          owner     = owner
          id        = "default"
        amount      = quantity
        locked      = locked
        lockContext = lockContext
        metadata    = metadata

      getInstrument_H = InstrumentId with
        depository = issuer
        issuer     = issuer
        id         = assetId
        version    = "0"

      getAccount = AccountId with
        custodian = issuer
        owner     = owner
        id        = "default"

      getAmount_H    = quantity
      isLocked       = locked
      getLockContext = lockContext

    choice Split : (ContractId Holding, ContractId Holding)
      with
        splitAmount   : Decimal
        splitMetadata : Metadata
      controller owner
      do
        assertMsg "Cannot split locked holding" (not locked)
        assertMsg "Split amount must be positive" (splitAmount > 0.0)
        assertMsg "Split amount too large" (splitAmount < quantity)
        h1 <- create this with quantity = splitAmount; metadata = splitMetadata
        h2 <- create this with quantity = quantity - splitAmount; metadata = splitMetadata
        return (toInterfaceContractId h1, toInterfaceContractId h2)

    choice Merge : ContractId Holding
      with
        otherHoldingCid : ContractId Holding
        mergeMetadata   : Metadata
      controller owner
      do
        other <- fetch otherHoldingCid
        assertMsg "Cannot merge locked holdings" (not locked && not (isLocked other))
        let selfAsInterface = toInterface @Holding this
        assertMsg "Can only merge same asset"   (getInstrument_H other == getInstrument_H selfAsInterface)
        assertMsg "Can only merge same account" (getAccount other == getAccount selfAsInterface)
        archive otherHoldingCid
        merged <- create this with quantity = quantity + getAmount_H other; metadata = mergeMetadata
        return (toInterfaceContractId merged)

    -- ✅ SECTION A.3: INITIATE CLAWBACK
    -- Controller: issuer AND compliance (dual-party initiation).
    -- Neither party can unilaterally freeze assets — both must co-sign.
    --
    -- With both issuer + compliance as controllers, and owner as signatory,
    -- all three parties' authority is present in the choice body.
    -- This allows us to archive the holding and create LockedHolding_Impl
    -- (which requires signatory owner + issuer + compliance) without any
    -- missing authorization errors.
    --
    -- This is stronger governance than most real-world regulated custodians,
    -- where a single compliance officer can often freeze unilaterally.
    nonconsuming choice InitiateClawback : ContractId ClawbackRequest
      with
        legalReason  : Text
        jurisdiction : Text
        caseReference: Text
      controller issuer, compliance   -- dual-party: neither can act alone
      do
        assertMsg "Cannot clawback a locked holding (resolve existing lock first)" (not locked)

        -- All three parties' authority is now in scope:
        --   owner      (signatory of this contract)
        --   issuer     (controller of this choice)
        --   compliance (controller of this choice)
        -- This satisfies LockedHolding_Impl's three-party signatory requirement,
        -- and also satisfies ClawbackRequest's three-party signatory requirement.
        archive self

        lockedHolding <- create LockedHolding_Impl with
          owner       = owner
          assetId     = assetId
          quantity    = quantity
          issuer      = issuer
          compliance  = compliance
          lockContext = "Clawback pending: " <> caseReference
          metadata    = addMetadata "clawback-initiated-by" (show issuer) metadata

        create ClawbackRequest with
          targetOwner      = owner
          assetId          = assetId
          quantity         = quantity
          issuer           = issuer
          compliance       = compliance
          legalReason      = legalReason
          jurisdiction     = jurisdiction
          caseReference    = caseReference
          lockedHoldingCid = lockedHolding
          metadata         = addMetadata "clawback-initiated-by" (show issuer) metadata


----------------------------------------------------------------------
-- LOCKED_HOLDING_IMPL
-- Clawback-pending state. Three-party signatory: owner + issuer + compliance.
-- Only ever created by InitiateClawback. Archived by AuthorizeClawback,
-- RejectClawback, or WithdrawClawback — never directly by any party.
----------------------------------------------------------------------

template LockedHolding_Impl
  with
    owner      : Party
    assetId    : Text
    quantity   : Decimal
    issuer     : Party
    compliance : Party
    lockContext: Text
    metadata   : Metadata
  where
    signatory owner, issuer, compliance

    ensure quantity > 0.0

    interface instance Holding for LockedHolding_Impl where
      view = HoldingView with
        instrument = InstrumentId with
          depository = issuer
          issuer     = issuer
          id         = assetId
          version    = "0"
        account = AccountId with
          custodian = issuer
          owner     = owner
          id        = "default"
        amount      = quantity
        locked      = True
        lockContext = Some lockContext
        metadata    = metadata

      getInstrument_H = InstrumentId with
        depository = issuer
        issuer     = issuer
        id         = assetId
        version    = "0"

      getAccount = AccountId with
        custodian = issuer
        owner     = owner
        id        = "default"

      getAmount_H    = quantity
      isLocked       = True
      getLockContext = Some lockContext


----------------------------------------------------------------------
-- CLAWBACK_REQUEST
-- Dual-auth workflow. References LockedHolding_Impl.
--
-- ✅ FIX (Option A): targetOwner is now a signatory, not merely an observer.
--
-- RATIONALE: LockedHolding_Impl requires signatory owner + issuer + compliance.
-- To archive it inside AuthorizeClawback / RejectClawback / WithdrawClawback,
-- all three parties' authority must be present in the choice execution context.
-- Making targetOwner a signatory embeds their authority at creation time.
--
-- This is legally correct: a clawback proceeding is a three-party legal event.
-- The owner is a named party — not just a bystander — and their node should
-- permanently hold the proceeding record as a stakeholder (not a pruneable observer).
--
-- Alice gains NO additional power: she does not control any choice. She cannot
-- block AuthorizeClawback (controller: compliance) or WithdrawClawback
-- (controller: issuer). Her authority is only present because InitiateClawback
-- runs inside Holding_Impl where she is already a signatory — she submits
-- no additional transaction to create ClawbackRequest.
--
-- CIP-0056 A.3 compliance is preserved: dual-party initiation (issuer + compliance)
-- is still required. Alice cannot unilaterally initiate or block the clawback.
----------------------------------------------------------------------

template ClawbackRequest
  with
    targetOwner      : Party
    assetId          : Text
    quantity         : Decimal
    issuer           : Party
    compliance       : Party
    legalReason      : Text
    jurisdiction     : Text
    caseReference    : Text
    lockedHoldingCid : ContractId LockedHolding_Impl
    metadata         : Metadata
  where
    signatory issuer, compliance, targetOwner  -- ✅ Option A: all three parties are formal participants
                                               -- targetOwner authority propagates into all choice bodies,
                                               -- enabling archive of LockedHolding_Impl (3-party signatory)

    -- Compliance formally authorizes with court order hash on record.
    -- Archives LockedHolding_Impl and creates recovered Holding_Impl for issuer.
    choice AuthorizeClawback : ClawbackResult
      with
        complianceOfficer : Text
        authorizationDate : Time
        courtOrderHash    : Text
      controller compliance
      do
        archive lockedHoldingCid

        recoveredHolding <- create Holding_Impl with
          owner       = issuer
          assetId     = assetId
          quantity    = quantity
          issuer      = issuer
          compliance  = compliance
          locked      = False
          lockContext = None
          metadata    = Map.fromList
            [ ("clawback-reason",           legalReason)
            , ("clawback-jurisdiction",     jurisdiction)
            , ("clawback-case",             caseReference)
            , ("clawback-authorized-by",    complianceOfficer)
            , ("clawback-court-order-hash", courtOrderHash)
            , ("clawback-date",             show authorizationDate)
            ]
          observers   = [] 

        return ClawbackResult with
          recoveredHoldingCid = toInterfaceContractId recoveredHolding
          clawbackMetadata    = metadata

    -- Compliance rejects: no valid legal basis.
    -- Archives LockedHolding_Impl and restores Alice's holding unlocked.
    choice RejectClawback : ContractId Holding
      with
        rejectionReason : Text
      controller compliance
      do
        archive lockedHoldingCid

        restored <- create Holding_Impl with
          owner       = targetOwner
          assetId     = assetId
          quantity    = quantity
          issuer      = issuer
          compliance  = compliance
          locked      = False
          lockContext = None
          metadata    = Map.fromList
            [ ("clawback-rejected-reason", rejectionReason)
            , ("clawback-case",            caseReference)
            ]
          observers   = [] 
        return (toInterfaceContractId restored)

    -- Either party can withdraw (made in error).
    -- Restores Alice's holding unlocked.
    choice WithdrawClawback : ContractId Holding
      controller issuer
      do
        archive lockedHoldingCid

        restored <- create Holding_Impl with
          owner       = targetOwner
          assetId     = assetId
          quantity    = quantity
          issuer      = issuer
          compliance  = compliance
          locked      = False
          lockContext = None
          metadata    = Map.fromList
            [ ("clawback-withdrawn", "true")
            , ("clawback-case",      caseReference)
            ]
          observers   = [] 
        return (toInterfaceContractId restored)


data ClawbackResult = ClawbackResult
  with
    recoveredHoldingCid : ContractId Holding
    clawbackMetadata    : Metadata
  deriving (Eq, Show)

          observers   = [] 
        
        let redeemerHash = sha256 (show redeemer)
        let txHash = sha256 (show redeemer <> show quantity <> show now)
        
        regulatorViewCid <- create RegulatorView with
          regulator = regulatorParty
          assetId = assetId
          quantity = quantity
          buyerHash = "BURN-DESTINATION"
          sellerHash = redeemerHash
          issuer = issuer
          executedAt = now
          eventType = "REDEMPTION"
          transactionHash = txHash
        
        return AuditedRedemptionResult with
          cashHolding = toInterfaceContractId cashHolding
          regulatorView = regulatorViewCid
          redemptionTime = now
    
    choice RejectRedemption : ContractId Holding
      with rejectionReason : Text
      controller issuer
      do
        let concreteAllocCid = coerceContractId @Allocation @Allocation_Impl allocationCid
        unlockedHolding <- exercise concreteAllocCid WithdrawAllocation with
          reason = rejectionReason
          withdrawMetadata = Map.fromList
            [ ("cantonsuite.io/operation", "redemption-rejected")
            , ("cantonsuite.io/rejection-reason", rejectionReason)
            , (cip0056Reason, rejectionReason)
            ]
        
        return unlockedHolding
    
    choice WithdrawRedemption : ContractId Holding
      controller redeemer
      do
        let concreteAllocCid = coerceContractId @Allocation @Allocation_Impl allocationCid
        unlockedHolding <- exercise concreteAllocCid WithdrawAllocation with
          reason = "Redeemer cancelled redemption"
          withdrawMetadata = Map.fromList
            [ ("cantonsuite.io/operation", "redemption-withdrawn")
            , (cip0056Reason, "Redeemer withdrew redemption request")
            ]
        
        return unlockedHolding

-- INITIATE REDEMPTION (Single Redemption) - ✅ ENHANCED WITH SYNC CHECK
-------------------------------

template InitiateRedemption
  with
    redeemer      : Party
    issuer        : Party
    compliance    : Party
    regulatorParty: Party
    assetId       : Text
    quantity      : Decimal
    reason        : Text
  where
    signatory redeemer, issuer
    observer compliance, regulatorParty
    
    -- ✅ ENHANCED: Added synchronizer compatibility check
    choice ExecuteRedemptionFlow : Either Text (ContractId RedemptionWorkflow)
      with
        holdingCid           : ContractId Holding
        deadlineHours        : Int
        miningRoundCid       : Optional (ContractId MiningRoundReference)
        redeemerCurrentSync  : Text
        targetSync           : Text
      controller redeemer, issuer
      do
        now <- getTime
        let deadline = addRelTime now (hours deadlineHours)
        
        -- ✅ Synchronizer compatibility check
        let needsMigration = redeemerCurrentSync /= targetSync
        
        if needsMigration then
          return (Left ("Cannot initiate redemption - synchronizer migration required: Redeemer on '" <> 
                       redeemerCurrentSync <> "' must migrate to '" <> targetSync <> 
                       "' before redemption. Please complete migration before attempting redemption."))
        else do
          -- ✅ All checks passed - proceed with redemption
          
          case miningRoundCid of
            Some roundCid -> do
              -- ✅ FIXED: Pass caller = issuer
              isValid <- exercise roundCid IsRoundValid with caller = issuer
              assertMsg "Canton Coin mining round expired before redemption start" isValid
              
              hasTime <- exercise roundCid HasTimeForExternalSigning with
                estimatedSigningDelaySeconds = 300
                caller = issuer
              assertMsg "Not enough time in mining round for redemption flow" hasTime
            None -> pure ()
          
          holding <- fetch holdingCid
          assertMsg "Insufficient balance" (getAmount_H holding >= quantity)
          assertMsg "Asset mismatch" ((getInstrument_H holding).id == assetId)
          assertMsg "Holding is locked" (not (isLocked holding))
          
          redemptionReq <- create RedemptionRequest with
            redeemer = redeemer
            issuer = issuer
            compliance = compliance
            regulatorParty = regulatorParty
            assetId = assetId
            quantity = quantity
            reason = reason
            requestedAt = now
            deadline = deadline
            miningRoundCid = miningRoundCid
          
          allocationReq <- exercise redemptionReq CreateRedemptionAllocation
          
          let concreteReqCid = coerceContractId @AllocationRequest @AllocationRequest_Impl allocationReq
          allocationInstruction <- exercise concreteReqCid AcceptRequest with
            holdingCid = holdingCid
            acceptMetadata = Map.fromList
              [ ("cantonsuite.io/operation", "redemption-allocation")
              , ("cantonsuite.io/reason", reason)
              , ("cantonsuite.io/synchronizer", targetSync)
              ]
          
          handler <- create AllocationExecutionHandler with
            registry = issuer
            complianceParty = compliance
          
          allocationCid <- exercise handler ExecuteAllocationInstruction with
            instructionCid = allocationInstruction
          
          workflowCid <- create RedemptionWorkflow with
            redeemer = redeemer
            issuer = issuer
            compliance = compliance
            regulatorParty = regulatorParty
            assetId = assetId
            quantity = quantity
            reason = reason
            requestedAt = now
            allocationCid = allocationCid
            cashPaymentPending = True
            miningRoundCid = miningRoundCid
          
          return (Right workflowCid)

template RedemptionEffect
  with
    redeemer       : Party
    issuer         : Party
    compliance     : Party
    regulatorParty : Party
    assetId        : Text
    quantity       : Decimal
    reason         : Text
    holdingCid     : ContractId Holding
    deadlineHours  : Int
    miningRoundCid : Optional (ContractId MiningRoundReference)
    createdAt      : Time
    effectId       : Text
  where
    signatory redeemer  -- ✅ Redeemer signs their own effect
    observer issuer, compliance
    
    ensure quantity > 0.0
    
    -- ✅ Issuer processes the effect (has both parties' authority)
    choice ExecuteRedemptionEffect : ContractId RedemptionWorkflow
      controller issuer
      do
        now <- getTime
        
        -- Verify holding ownership
        holding <- fetch holdingCid
        let account = getAccount holding
        assertMsg "Wrong holding owner" (account.owner == redeemer)
        assertMsg "Insufficient balance" (getAmount_H holding >= quantity)
        assertMsg "Asset mismatch" ((getInstrument_H holding).id == assetId)
        
        -- Create InitiateRedemption with both parties' authority
        initiator <- create InitiateRedemption with
          redeemer = redeemer
          issuer = issuer
          compliance = compliance
          regulatorParty = regulatorParty
          assetId = assetId
          quantity = quantity
          reason = reason
        
        -- Execute the flow (both parties already have authority)
        -- Note: This returns Either, we need to unwrap it
        result <- exercise initiator ExecuteRedemptionFlow with
          holdingCid = holdingCid
          deadlineHours = deadlineHours
          miningRoundCid = miningRoundCid
          redeemerCurrentSync = "global"  -- Default to global for effect-based
          targetSync = "global"
        
        case result of
          Right workflowCid -> return workflowCid
          Left errorMsg -> error errorMsg
    
    choice CancelRedemptionEffect : ()
      controller redeemer
      do pure ()

template BatchRedemptionRequest
  with
    issuer         : Party
    compliance     : Party
    regulatorParty : Party
    assetId        : Text
    redemptions    : [(Party, Decimal, Text)]  -- (redeemer, quantity, reason)
    requestedAt    : Time
  where
    signatory issuer
    observer compliance, regulatorParty
    
    -- ✅ NEW: Process pre-created effects (consent already collected)
    choice ProcessRedemptionEffects : [ContractId RedemptionWorkflow]
      with effectCids : [ContractId RedemptionEffect]
      controller issuer
      do
        assertMsg "Effect count must match redemption count" 
          (length effectCids == length redemptions)
        
        -- Verify each effect matches the redemption list
        effects <- mapA fetch effectCids
        forA (zip redemptions effects) $ \((redeemer, qty, _), effect) -> do
          assertMsg "Effect redeemer mismatch" (effect.redeemer == redeemer)
          assertMsg "Effect quantity mismatch" (effect.quantity == qty)
          assertMsg "Effect asset mismatch" (effect.assetId == assetId)
        
        -- Process all effects
        forA effectCids $ \effectCid ->
          exercise effectCid ExecuteRedemptionEffect
    
    -- ✅ DEPRECATED: Old direct processing (kept for backward compatibility)
    -- NOTE: This will fail with authorization errors - use ProcessRedemptionEffects instead
    choice ProcessBatchRedemptions_DEPRECATED : [ContractId RedemptionWorkflow]
      with
        holdingCids     : [ContractId Holding]
        deadlineHours   : Int
        miningRoundCid  : Optional (ContractId MiningRoundReference)
      controller issuer
      do
        error "DEPRECATED: Use ProcessRedemptionEffects instead. Redeemers must create RedemptionEffect contracts first."

-- Redemption Queue (Updated for Effects)
-------------------------------

template RedemptionQueue
  with
    issuer          : Party
    assetId         : Text
    pendingEffects  : [ContractId RedemptionEffect]  -- ✅ Changed from workflows
    maxBatchSize    : Int
  where
    signatory issuer
    
    ensure maxBatchSize > 0
    
    nonconsuming choice AddToQueue : ContractId RedemptionQueue
      with effectCid : ContractId RedemptionEffect
      controller issuer
      do
        archive self
        create this with
          pendingEffects = pendingEffects ++ [effectCid]
    
    choice ProcessQueueBatch : ContractId RedemptionQueue
      with cashAmounts : [Decimal]
      controller issuer
      do
        let batchSize = min maxBatchSize (length pendingEffects)
        let (toBatch, remaining) = splitAt batchSize pendingEffects
        
        assertMsg "Cash amounts mismatch" (length cashAmounts >= batchSize)
        
        -- Execute effects to get workflows
        workflowCids <- forA toBatch $ \effectCid ->
          exercise effectCid ExecuteRedemptionEffect
        
        -- Complete redemptions with cash payments
        forA (zip workflowCids cashAmounts) $ \(workflowCid, cashAmount) -> do
          exercise workflowCid CompleteRedemptionAtomic with
            cashAmount = cashAmount
        
         
        create this with
          pendingEffects = remaining

template AllocationInstruction_Impl
  with
    holder           : Party
    instrument       : InstrumentId
    amount           : Decimal
    settlementId     : Text
    deadline         : Time
    settlementApp    : Party
    metadata         : Metadata
    sourceHoldingCid : ContractId Holding
  where
    signatory holder
    observer instrument.issuer, settlementApp
    
    ensure amount > 0.0
    
    interface instance AllocationInstruction for AllocationInstruction_Impl where
      view = AllocationInstructionView with
        holder = holder
        instrument = instrument
        amount = amount
        settlementId = settlementId
        deadline = deadline
        settlementApp = settlementApp
        metadata = metadata
      
      getHolder_AI = holder
      getInstrument_AI = instrument
      getAmount_AI = amount
      getSettlementId_AI = settlementId
      getDeadline_AI = deadline
      getSettlementApp_AI = settlementApp
    
    choice ExecuteInstruction : ContractId Allocation
      with
        executionMetadata : Metadata
      controller instrument.issuer
      do
        now <- getTime
        assertMsg "Deadline passed" (now <= deadline)
        
        sourceHolding <- fetch sourceHoldingCid
        assertMsg "Insufficient funds" (getAmount_H sourceHolding >= amount)
        assertMsg "Instrument mismatch" (getInstrument_H sourceHolding == instrument)
        assertMsg "Already locked" (not (isLocked sourceHolding))

        -- Read compliance from source holding
        let concreteCid = coerceContractId @Holding @Holding_Impl sourceHoldingCid
        concreteHolding <- fetch concreteCid
        let complianceParty = concreteHolding.compliance
        
        archive sourceHoldingCid
        
        let (lockedAmount, remainderAmount) = 
              if getAmount_H sourceHolding > amount
              then (amount, Some (getAmount_H sourceHolding - amount))
              else (amount, None)
        
        case remainderAmount of
          Some remainder -> do
            _ <- create Holding_Impl with
              owner = holder
              assetId = instrument.id
              quantity = remainder
              issuer = instrument.issuer
              compliance = complianceParty
              locked = False
              lockContext = None
              metadata = emptyMetadata
              observers = []
            pure ()
          None -> pure ()
        
        lockedHolding <- create Holding_Impl with
          owner = holder
          assetId = instrument.id
          quantity = lockedAmount
          issuer = instrument.issuer
          compliance = complianceParty
          locked = True
          lockContext = Some ("Settlement: " <> settlementId)
          metadata = addMetadata cip0056LockContext settlementId executionMetadata
          observers = []

        allocation <- create Allocation_Impl with
          owner = holder
          instrument = instrument
          amount = amount
          settlementId = settlementId
          deadline = deadline
          settlementApp = settlementApp
          metadata = executionMetadata
          lockedHoldingCid = toInterfaceContractId lockedHolding
        
        return (toInterfaceContractId allocation)
    
    choice AbortInstruction : ()
      with
        reason        : Text
        abortMetadata : Metadata
      controller instrument.issuer
      do 
        assertMsg "Must provide reason" (reason /= "")
        let finalMetadata = addMetadata cip0056Reason reason abortMetadata
        pure ()
    
    choice WithdrawInstruction : ()
      with
        reason            : Text
        withdrawMetadata  : Metadata
      controller holder
      do 
        assertMsg "Must provide reason" (reason /= "")
        let finalMetadata = addMetadata cip0056Reason reason withdrawMetadata
        pure ()

template DVPSettlement
  with
    settlementId    : Text
    settlementApp   : Party
    buyer           : Party
    seller          : Party
    buyerAsset      : InstrumentId
    sellerAsset     : InstrumentId
    buyerAmount     : Decimal
    sellerAmount    : Decimal
    deadline        : Time
    complianceParty : Party
    regulatorParty  : Party
    description     : Text
  where
    signatory settlementApp
    observer buyer, seller, complianceParty, regulatorParty
    
    ensure
      buyerAmount > 0.0 &&
      sellerAmount > 0.0
    
    nonconsuming choice RequestAllocations : (ContractId AllocationRequest, ContractId AllocationRequest)
      controller settlementApp
      do
        now <- getTime
        assertMsg "Settlement already expired" (now < deadline)
        
        buyerReq <- create AllocationRequest_Impl with
          holder = buyer
          instrument = buyerAsset
          amount = buyerAmount
          settlementId = settlementId
          deadline = deadline
          settlementApp = settlementApp
          metadata = Map.fromList
            [ ("cantonsuite.io/settlement-id", settlementId)
            , ("cantonsuite.io/counterparty", show seller)
            , ("cantonsuite.io/role", "buyer")
            , ("cantonsuite.io/description", description)
            ]
        
        sellerReq <- create AllocationRequest_Impl with
          holder = seller
          instrument = sellerAsset
          amount = sellerAmount
          settlementId = settlementId
          deadline = deadline
          settlementApp = settlementApp
          metadata = Map.fromList
            [ ("cantonsuite.io/settlement-id", settlementId)
            , ("cantonsuite.io/counterparty", show buyer)
            , ("cantonsuite.io/role", "seller")
            , ("cantonsuite.io/description", description)
            ]
        
        return (toInterfaceContractId buyerReq, toInterfaceContractId sellerReq)
    
    -- ✅ NEW: Synchronizer-aware allocation request
    nonconsuming choice RequestAllocationsWithSyncCheck : Either Text (ContractId AllocationRequest, ContractId AllocationRequest)
      with
        buyerCurrentSync  : Text
        sellerCurrentSync : Text
        targetSync        : Text
      controller settlementApp
      do
        now <- getTime
        assertMsg "Settlement already expired" (now < deadline)
        
        -- Check if migration is needed
        let needsBuyerMigration = buyerCurrentSync /= targetSync
        let needsSellerMigration = sellerCurrentSync /= targetSync
        
        if needsBuyerMigration || needsSellerMigration then do
          let buyerMsg = if needsBuyerMigration 
                        then "Buyer on " <> buyerCurrentSync <> " (needs migration to " <> targetSync <> ")"
                        else ""
          let sellerMsg = if needsSellerMigration
                         then "Seller on " <> sellerCurrentSync <> " (needs migration to " <> targetSync <> ")"
                         else ""
          let fullMsg = "Synchronizer migration required before settlement: " <> buyerMsg <> 
                       (if needsBuyerMigration && needsSellerMigration then "; " else "") <> 
                       sellerMsg
          return (Left fullMsg)
        else do
          -- All parties on same sync - proceed normally
          buyerReq <- create AllocationRequest_Impl with
            holder = buyer
            instrument = buyerAsset
            amount = buyerAmount
            settlementId = settlementId
            deadline = deadline
            settlementApp = settlementApp
            metadata = Map.fromList
              [ ("cantonsuite.io/settlement-id", settlementId)
              , ("cantonsuite.io/counterparty", show seller)
              , ("cantonsuite.io/role", "buyer")
              , ("cantonsuite.io/synchronizer", targetSync)
              , ("cantonsuite.io/description", description)
              ]
          
          sellerReq <- create AllocationRequest_Impl with
            holder = seller
            instrument = sellerAsset
            amount = sellerAmount
            settlementId = settlementId
            deadline = deadline
            settlementApp = settlementApp
            metadata = Map.fromList
              [ ("cantonsuite.io/settlement-id", settlementId)
              , ("cantonsuite.io/counterparty", show buyer)
              , ("cantonsuite.io/role", "seller")
              , ("cantonsuite.io/synchronizer", targetSync)
              , ("cantonsuite.io/description", description)
              ]
          
          return (Right (toInterfaceContractId buyerReq, toInterfaceContractId sellerReq))
    
    choice ExecuteAtomicDVP : AtomicDVPResult
      with
        buyerAllocationCid  : ContractId Allocation
        sellerAllocationCid : ContractId Allocation
      controller settlementApp, buyer, seller
      do
        now <- getTime
        assertMsg "Settlement deadline passed" (now <= deadline)
        
        buyerAlloc <- fetch buyerAllocationCid
        sellerAlloc <- fetch sellerAllocationCid
        
        assertMsg "Buyer allocation mismatch" (getAmount_A buyerAlloc >= buyerAmount)
        assertMsg "Seller allocation mismatch" (getAmount_A sellerAlloc >= sellerAmount)
        assertMsg "Buyer allocation expired" (now <= getDeadline_A buyerAlloc)
        assertMsg "Seller allocation expired" (now <= getDeadline_A sellerAlloc)
        
        let buyerConcreteCid = coerceContractId @Allocation @Allocation_Impl buyerAllocationCid
        let sellerConcreteCid = coerceContractId @Allocation @Allocation_Impl sellerAllocationCid
        
        buyerConcreteAlloc <- fetch buyerConcreteCid
        sellerConcreteAlloc <- fetch sellerConcreteCid
        
        archive buyerConcreteAlloc.lockedHoldingCid
        
        if getAmount_A buyerAlloc > buyerAmount then do
          _ <- create Holding_Impl with
            owner = buyer
            assetId = buyerAsset.id
            quantity = getAmount_A buyerAlloc - buyerAmount
            issuer = buyerAsset.issuer
            compliance = complianceParty
            locked = False
            lockContext = None
            metadata = emptyMetadata
            observers = [] 
          pure ()
        else pure ()
        
        archive sellerConcreteAlloc.lockedHoldingCid
        
        if getAmount_A sellerAlloc > sellerAmount then do
          _ <- create Holding_Impl with
            owner = seller
            assetId = sellerAsset.id
            quantity = getAmount_A sellerAlloc - sellerAmount
            issuer = sellerAsset.issuer
            compliance = complianceParty
            locked = False
            lockContext = None
            metadata = emptyMetadata
            observers = [] 
          pure ()
        else pure ()
        
        sellerPaymentHolding <- create Holding_Impl with
          owner = seller
          assetId = buyerAsset.id
          quantity = buyerAmount
          issuer = buyerAsset.issuer
          compliance = complianceParty
          locked = False
          lockContext = None
          metadata = Map.fromList
            [ ("cantonsuite.io/settlement-id", settlementId)
            , ("cantonsuite.io/execution-time", show now)
            , ("cantonsuite.io/transfer-type", "DVP-payment")
            , ("cantonsuite.io/description", description)
            ]
          observers = []
        
        buyerAssetHolding <- create Holding_Impl with
          owner = buyer
          assetId = sellerAsset.id
          quantity = sellerAmount
          issuer = sellerAsset.issuer
          compliance = complianceParty
          locked = False
          lockContext = None
          metadata = Map.fromList
            [ ("cantonsuite.io/settlement-id", settlementId)
            , ("cantonsuite.io/execution-time", show now)
            , ("cantonsuite.io/transfer-type", "DVP-delivery")
            , ("cantonsuite.io/description", description)
            ]
          observers = []
        
        archive buyerAllocationCid
        archive sellerAllocationCid
        
        return AtomicDVPResult with
          sellerReceivedPayment = toInterfaceContractId sellerPaymentHolding
          buyerReceivedAsset = toInterfaceContractId buyerAssetHolding
          executedAt = now
    
    choice CancelSettlement : ()
      with
        reason : Text
      controller settlementApp
      do 
        assertMsg "Must provide reason" (reason /= "")
        pure ()
    
    choice ExtendDeadline : ContractId DVPSettlement
      with
        newDeadline : Time
      controller settlementApp, buyer, seller
      do
        now <- getTime
        assertMsg "New deadline must be in future" (newDeadline > now)
        create this with deadline = newDeadline

        assertMsg "Holding is locked" (not (isLocked senderHolding))

        create TransferExecutionRequest with
          sender = sender
          receiver = receiver
          instrument = instrument
          amount = amount
          senderHoldingCid = senderHoldingCid
          instructionCid = toInterfaceContractId self
          executionMetadata = executionMetadata

    choice Abort : ()
      with
        reason        : Text
        abortMetadata : Metadata
      controller instrument.issuer
      do
        _ <- create this with
          status = Aborted
          metadata = addMetadata cip0056Reason reason abortMetadata
        pure ()

    choice Withdraw : ()
      with
        reason           : Text
        withdrawMetadata : Metadata
      controller sender
      do
        assertMsg "Can only withdraw pending transfers" (status == Pending)
        _ <- create this with
          status = Withdrawn
          metadata = addMetadata cip0056Reason reason withdrawMetadata
        pure ()

-------------------------------
-- Transfer Execution Request
--
-- KYC design:
--   - SENDER voucher: verified by registry (compliance co-controller) in ExecuteTransfer ✅
--   - RECEIVER voucher: NOT verified here — compliance already vetted both parties ✅
--   - AcceptTransfer is purely Bob saying "yes, I accept these tokens" ✅
-------------------------------

template TransferExecutionRequest
  with
    sender            : Party
    receiver          : Party
    instrument        : InstrumentId
    amount            : Decimal
    senderHoldingCid  : ContractId Holding
    instructionCid    : ContractId TransferInstruction
    executionMetadata : Metadata
  where
    signatory instrument.issuer, sender
    observer receiver

    -- ✅ No voucher parameter — compliance check happened in ExecuteTransfer.
    --    Bob just accepts the transfer.
    choice AcceptTransfer : TransferResult
      with
      controller receiver, instrument.issuer 
      do
        senderHolding <- fetch senderHoldingCid

        let concreteSenderCid = coerceContractId @Holding @Holding_Impl senderHoldingCid
        concreteSender <- fetch concreteSenderCid

        let complianceParty = concreteSender.compliance

        assertMsg "Insufficient funds"  (getAmount_H senderHolding >= amount)
        assertMsg "Instrument mismatch" (getInstrument_H senderHolding == instrument)
        assertMsg "Holding is locked"   (not (isLocked senderHolding))

        archive senderHoldingCid

        if getAmount_H senderHolding > amount then do
          let senderAccount = getAccount senderHolding
          _ <- create Holding_Impl with
            owner       = senderAccount.owner
            assetId     = instrument.id
            quantity    = getAmount_H senderHolding - amount
            issuer      = instrument.issuer
            compliance  = complianceParty
            locked      = False
            lockContext = None
            metadata    = emptyMetadata
            observers   = [] 
          pure ()
        else
          pure ()

        receiverHolding <- create Holding_Impl with
          owner       = receiver
          assetId     = instrument.id
          quantity    = amount
          issuer      = instrument.issuer
          compliance  = complianceParty
          locked      = False
          lockContext = None
          metadata    = executionMetadata
          observers   = [] 

        let concreteCid = coerceContractId @TransferInstruction @TransferInstruction_Impl instructionCid
        concreteInstruction <- fetch concreteCid
        archive concreteCid

        now <- getTime
        _ <- create concreteInstruction with
          status   = Executed
          metadata = addMetadata "execution-time" (show now) executionMetadata

        return TransferResult with
          receiverHoldingCid = toInterfaceContractId receiverHolding
          resultMetadata     = executionMetadata

    choice RejectTransfer : ()
      with
        reason : Text
      controller receiver, instrument.issuer
      do
        let concreteCid = coerceContractId @TransferInstruction @TransferInstruction_Impl instructionCid
        concreteInstruction <- fetch concreteCid
        archive concreteCid

        _ <- create concreteInstruction with
          status   = Aborted
          metadata = addMetadata cip0056Reason ("Rejected by receiver: " <> reason) concreteInstruction.metadata

        pure ()

        assertMsg "Asset mismatch"       ((getInstrument_H senderHolding).id == assetId)
        assertMsg "Holding is locked"    (not (isLocked senderHolding))

        let needsSenderMigration = senderCurrentSync /= targetSync
        let needsReceiverMigration = receiverCurrentSync /= targetSync

          let senderMsg = if needsSenderMigration
                         then "Sender on '" <> senderCurrentSync <> "' must migrate to '" <> targetSync <> "'"
                         else ""
          let receiverMsg = if needsReceiverMigration
                           then "Receiver on '" <> receiverCurrentSync <> "' must migrate to '" <> targetSync <> "'"
                           else ""

                       senderMsg <>
                       (if needsSenderMigration && needsReceiverMigration then "; " else "") <>
                       receiverMsg <>
                       ". Please complete migration before attempting transfer."
          return (Left fullMsg)
        else do
          -- Parties on same sync - proceed
          now <- getTime
          let deadline = addRelTime now (hours deadlineHours)

          instruction <- create TransferInstruction_Impl with
            sender   = sender
            receiver = receiver
            instrument = InstrumentId with

              , ("cantonsuite.io/synchronizer",  targetSync)
              , ("cantonsuite.io/notes",          notes)
              ]
            senderHoldingCid = senderHoldingCid

          return (Right (toInterfaceContractId instruction))

    -- ✅ EXISTING: Keep original InitiateFOPTransfer for backward compatibility
    choice InitiateFOPTransfer : ContractId TransferInstruction
      with

        deadlineHours    : Int
        notes            : Text

    -- - Maintains atomicity guarantees of DVP settlement
    -- - Enables seamless multi-synchronizer trading networks
    choice CreateDVPWithSyncCheck : Either Text (ContractId DVPSettlement)
      with
        paymentAssetId    : Text
        buyerCurrentSync  : Text  -- Current synchronizer where buyer's holdings reside
        sellerCurrentSync : Text  -- Current synchronizer where seller's holdings reside
        targetSync        : Text  -- Target synchronizer for settlement
      controller assetIssuer
      do
        now <- getTime
        assertMsg "Trade has expired" (now <= expiresAt)
        
        -- Check if synchronizers are compatible
        let needsBuyerMigration = buyerCurrentSync /= targetSync
        let needsSellerMigration = sellerCurrentSync /= targetSync
        
        if needsBuyerMigration || needsSellerMigration then do
          -- Return error with detailed migration instructions
          let buyerMsg = if needsBuyerMigration
                        then "Buyer on '" <> buyerCurrentSync <> "' must migrate to '" <> targetSync <> "'"
                        else ""
          let sellerMsg = if needsSellerMigration
                         then "Seller on '" <> sellerCurrentSync <> "' must migrate to '" <> targetSync <> "'"
                         else ""
          let separator = if needsBuyerMigration && needsSellerMigration then "; " else ""
          let fullMsg = "Cannot create DVP settlement - synchronizer migration required: " <>
                       buyerMsg <> separator <> sellerMsg <>
                       ". Please complete migration before attempting settlement."
          return (Left fullMsg)
        else do
          -- All parties on same synchronizer - proceed with DVP creation
          let settlementDeadline = addRelTime now (hours 48)
          let settlementId = "TRADE-SYNC-" <> show self <> "-" <> show now
          
          settlementCid <- create DVPSettlement with
            settlementId = settlementId
            settlementApp = assetIssuer
            buyer = buyer
            seller = seller
            buyerAsset = InstrumentId with
              depository = assetIssuer
              issuer = assetIssuer
              id = paymentAssetId
              version = "0"
            sellerAsset = InstrumentId with
              depository = assetIssuer
              issuer = assetIssuer
              id = assetId
              version = "0"
            buyerAmount = quantity * pricePerUnit
            sellerAmount = quantity
            deadline = settlementDeadline
            complianceParty = complianceParty
            regulatorParty = regulatorParty
            description = "Trade settlement (Sync: " <> targetSync <> "): " <> show quantity <> " " <> assetId
          
          return (Right settlementCid)
    
    -- ✅ PRESERVED: Original atomic settlement with all existing validations
    -- This choice performs the complete trade settlement in a single atomic transaction:
    -- - KYC verification for both parties
    -- - Mining round validation (for Canton Coin trades)
    -- - Holding verification and transfer
    -- - Change calculation and return
    -- - Atomic audit record creation
    -- - Regulatory reporting
    choice ExecuteAtomicSettlement : AuditedSettlementResult
      with
        buyerPaymentCid : ContractId Holding
        sellerAssetCid  : ContractId Holding
        miningRoundCid  : Optional (ContractId MiningRoundReference)
      controller assetIssuer, complianceParty
      do
        now <- getTime
        assertMsg "Trade has expired" (now <= expiresAt)
        
        -- KYC Verification
        let totalPrice = quantity * pricePerUnit
        exercise buyerVoucherCid VerifyKYC with
          tradeAmount = totalPrice
          tradingParty = buyer
        exercise sellerVoucherCid VerifyKYC with
          tradeAmount = quantity
          tradingParty = seller
        
        -- Mining Round Validation (for Canton Coin)
        case miningRoundCid of
          Some roundCid -> do
            isValid <- exercise roundCid IsRoundValid with caller = assetIssuer
            assertMsg "Canton Coin mining round expired" isValid
            hasTime <- exercise roundCid HasTimeForExternalSigning with
              estimatedSigningDelaySeconds = 120
              caller = assetIssuer
            assertMsg "Insufficient time remaining in mining round" hasTime
          None -> pure ()
        
        -- Holding Verification
        paymentHolding <- fetch buyerPaymentCid
        assetHolding <- fetch sellerAssetCid
        
        assertMsg "Payment insufficient" (getAmount_H paymentHolding >= totalPrice)
        assertMsg "Asset quantity insufficient" (getAmount_H assetHolding >= quantity)
        assertMsg "Payment from wrong party" (getOwner_H paymentHolding == buyer)
        assertMsg "Asset from wrong party" (getOwner_H assetHolding == seller)
        
        -- Archive consumed contracts
        archive buyerVoucherCid
        archive sellerVoucherCid
        archive buyerPaymentCid
        archive sellerAssetCid
        
        -- Calculate change
        let paymentChange = getAmount_H paymentHolding - totalPrice
        let assetChange = getAmount_H assetHolding - quantity
        
        -- Return change to buyer (if any)
        when (paymentChange > 0.0) $ do
          _ <- create Holding_Impl with
            owner = buyer
            assetId = (getInstrument_H paymentHolding).id
            quantity = paymentChange
            issuer = (getInstrument_H paymentHolding).issuer
            compliance = complianceParty
            locked = False
            lockContext = None
            metadata = Map.empty
            observers   = [] 
          pure ()
        
        -- Return change to seller (if any)
        when (assetChange > 0.0) $ do
          _ <- create Holding_Impl with
            owner = seller
            assetId = (getInstrument_H assetHolding).id
            quantity = assetChange
            issuer = (getInstrument_H assetHolding).issuer
            compliance = complianceParty
            locked = False
            lockContext = None
            metadata = Map.empty
            observers   = [] 
          pure ()
        
        -- Create seller's payment holding (buyer's payment → seller)
        sellerPaymentHolding <- create Holding_Impl with
          owner = seller
          assetId = (getInstrument_H paymentHolding).id
          quantity = totalPrice
          issuer = (getInstrument_H paymentHolding).issuer
          compliance = complianceParty
          locked = False
          lockContext = None
          metadata = Map.fromList
            [ ("cantonsuite.io/trade-id", show self)
            , ("cantonsuite.io/execution-time", show now)
            ]
          observers   = [] 
        
        -- Create buyer's asset holding (seller's asset → buyer)
        buyerAssetHolding <- create Holding_Impl with
          owner = buyer
          assetId = (getInstrument_H assetHolding).id
          quantity = quantity
          issuer = (getInstrument_H assetHolding).issuer
          compliance = complianceParty
          locked = False
          lockContext = None
          metadata = Map.fromList
            [ ("cantonsuite.io/trade-id", show self)
            , ("cantonsuite.io/execution-time", show now)
            ]
          observers   = [] 
        
        -- ✅ ATOMIC AUDIT: Create RegulatorView with correct field names
        let tradeHash = sha256 (show self <> show now)
        
        auditRecord <- create RegulatorView with
          regulator = regulatorParty
          assetId = assetId
          quantity = quantity
          buyerHash = sha256 (show buyer)
          sellerHash = sha256 (show seller)
          issuer = assetIssuer
          executedAt = now
          eventType = "TRADE"
          transactionHash = tradeHash
        
        -- ✅ Return AuditedSettlementResult with correct field names
        return AuditedSettlementResult with
          buyerHolding = toInterfaceContractId buyerAssetHolding
          sellerHolding = toInterfaceContractId sellerPaymentHolding
          regulatorView = auditRecord
          settlementTime = now
    
    -- ✅ PRESERVED: Trade expiration handling
    choice ExpireTrade : ()
      controller assetIssuer
      do
        now <- getTime
        assertMsg "Trade has not expired yet" (now > expiresAt)
        pure ()
